41class SPIFInputDevice( 

42 ApplicationFPGAVertex, PopulationApplicationVertex, 

43 AbstractSendMeMulticastCommandsVertex): 

44 """ A 1D input device connected to SpiNNaker using a SPIF board. 

45 """ 

46 

47 #: SPIF outputs to 8 FPGA output links, so we split all input into 8 

48 INPUT_MASK = 7 

49 

50 #: The number of devices in existence, to work out the key 

51 __n_devices = 0 

52 

53 __slots__ = [ 

54 "__spif_mask", 

55 "__index_by_slice", 

56 "__base_key", 

57 "__pipe", 

58 "__key_mask", 

59 "__m_vertex_mask", 

60 "__neuron_bits", 

61 "__index_bits", 

62 "__index_shift", 

63 "__input_mask", 

64 "__input_shift"] 

65 

66 def __init__(self, pipe: int, n_neurons: int, n_neurons_per_partition: int, 

67 base_key: Optional[int] = None, 

68 board_address: Optional[str] = None, 

69 chip_coords: Optional[Tuple[int, int]] = None): 

70 """ 

71 

72 :param pipe: Which pipe on SPIF the retina is connected to 

73 :param n_neurons: The number of neurons in the device 

74 :param n_neurons_per_partition: The number of neurons per partition 

75 :param base_key: 

76 The key that is common over the whole vertex, 

77 or None to use the pipe number as the key 

78 :param board_address: 

79 The IP address of the board to which the FPGA is connected, or None 

80 to use the default board or chip_coords. Note chip_coords will be 

81 used first if both are specified, with board_address then being 

82 used if the coordinates don't connect to an FPGA. 

83 :param chip_coords: 

84 The coordinates of the chip to which the FPGA is connected, or 

85 None to use the default board or board_address. Note chip_coords 

86 will be used first if board_address is also specified, with 

87 board_address then being used if the coordinates don't connect to 

88 an FPGA. 

89 """ 

90 # Do some checks 

91 if n_neurons_per_partition < self.INPUT_MASK: 

92 raise ConfigurationException( 

93 "The number of neurons per partition must not be <" 

94 f" {self.INPUT_MASK} ({n_neurons_per_partition} specified)") 

95 if not self.__is_power_of_2(n_neurons_per_partition): 

96 raise ConfigurationException( 

97 f"n_neurons_per_partition ({n_neurons_per_partition}) " 

98 "must be a power of 2") 

99 if pipe >= N_PIPES: 

100 raise ConfigurationException( 

101 f"Pipe {pipe} is bigger than maximum allowed {N_PIPES}") 

102 

103 n_machine_vertices = int( 

104 math.ceil(n_neurons / n_neurons_per_partition)) 

105 

106 # Call the super 

107 super().__init__( 

108 n_atoms=n_neurons, 

109 n_machine_vertices_per_link=n_machine_vertices, 

110 incoming_fpga_connections=self.__incoming_fpgas( 

111 board_address, chip_coords), 

112 outgoing_fpga_connection=self.__outgoing_fpga( 

113 board_address, chip_coords)) 

114 

115 # The mask is going to be made up of: 

116 # | K | N_I | N_0 | N_F | 

117 # K = base key 

118 # N_I = index of sub-partition 

119 # N_0 = 0s for neuron IDs not cared about 

120 # N_F = FPGA index 

121 # Now - go calculate: 

122 self.__neuron_bits = get_n_bits(n_neurons) 

123 self.__index_bits = get_n_bits(n_machine_vertices) 

124 self.__index_shift = get_n_bits(n_neurons_per_partition) 

125 

126 # Mask to apply to route packets at input 

127 n_key_bits = BITS_IN_KEY - self.__neuron_bits 

128 self.__key_mask = ((1 << n_key_bits) - 1) << self.__neuron_bits 

129 self.__spif_mask = self.__key_mask + self.INPUT_MASK 

130 sub_mask = (1 << self.__index_bits) - 1 

131 self.__m_vertex_mask = ( 

132 self.__key_mask + (sub_mask << self.__index_shift)) 

133 

134 # A dictionary to get vertex index from FPGA and slice 

135 self.__index_by_slice: Dict[Tuple[int, Slice], int] = dict() 

136 

137 self.__pipe = pipe 

138 if base_key is None: 

139 self.__base_key = SPIFInputDevice.__n_devices 

140 else: 

141 self.__base_key = base_key 

142 self.__base_key = self.__base_key << self.__neuron_bits 

143 SPIFInputDevice.__n_devices += 1 

144 

145 # Generate the shifts and masks to convert the SPIF Ethernet inputs to 

146 # output format 

147 self.__input_mask = (1 << self.__neuron_bits) - 1 

148 self.__input_shift = 0 

149 

150 def __is_power_of_2(self, v: int) -> bool: 

151 """ Determine if a value is a power of 2 

152 

153 :param v: The value to test 

154 """ 

155 return (v & (v - 1) == 0) and (v != 0) 

156 

157 def __incoming_fpgas( 

158 self, board_address: Optional[str], 

159 chip_coords: Optional[Tuple[int, int]]) -> List[FPGAConnection]: 

160 """ Get the incoming FPGA connections """ 

161 # We use every other odd link 

162 return [FPGAConnection(SPIF_FPGA_ID, i, board_address, chip_coords) 

163 for i in SPIF_INPUT_FPGA_LINKS] 

164 

165 def __outgoing_fpga( 

166 self, board_address: Optional[str], 

167 chip_coords: Optional[Tuple[int, int]]) -> FPGAConnection: 

168 """ Get the outgoing FPGA connection (for commands) """ 

169 return FPGAConnection( 

170 SPIF_FPGA_ID, SPIF_OUTPUT_FPGA_LINK, board_address, chip_coords) 

171 

172 def __fpga_index(self, fpga_link_id: int) -> int: 

173 # We use every other odd link, so we can work out the "index" of the 

174 # link in the list as follows 

175 return (fpga_link_id - 1) // 2 

176 

177 @overrides(ApplicationFPGAVertex.get_incoming_slice_for_link) 

178 def get_incoming_slice_for_link( 

179 self, link: FPGAConnection, index: int) -> Slice: 

180 vertex_slice = super().get_incoming_slice_for_link(link, index) 

181 self.__index_by_slice[link.fpga_link_id, vertex_slice] = index 

182 return vertex_slice 

183 

184 @overrides(ApplicationFPGAVertex.get_machine_fixed_key_and_mask) 

185 def get_machine_fixed_key_and_mask(self, machine_vertex: MachineVertex, 

186 partition_id: str) -> BaseKeyAndMask: 

187 assert isinstance(machine_vertex, MachineFPGAVertex) 

188 fpga_link_id = machine_vertex.fpga_link_id 

189 vertex_slice = machine_vertex.vertex_slice 

190 index = self.__index_by_slice[fpga_link_id, vertex_slice] 

191 key = self.__base_key | (index << self.__index_shift) 

192 

193 fpga = self.__fpga_index(fpga_link_id) 

194 

195 # Build the key from the components 

196 fpga_key = key + fpga 

197 fpga_mask = self.__m_vertex_mask | self.__spif_mask 

198 return BaseKeyAndMask(fpga_key, fpga_mask) 

199 

200 @overrides(ApplicationFPGAVertex.get_fixed_key_and_mask) 

201 def get_fixed_key_and_mask(self, partition_id: str) -> BaseKeyAndMask: 

202 return BaseKeyAndMask(self.__base_key, self.__key_mask) 

203 

204 @property 

205 @overrides(AbstractSendMeMulticastCommandsVertex.start_resume_commands) 

206 def start_resume_commands(self) -> Iterable[MultiCastCommand]: 

207 # Make sure everything has stopped 

208 commands = [SpiNNFPGARegister.STOP.cmd()] 

209 

210 # Clear the counters 

211 commands.append(SPIFRegister.OUT_PERIPH_PKT_CNT.cmd(0)) 

212 commands.append(SPIFRegister.CONFIG_PKT_CNT.cmd(0)) 

213 commands.append(SPIFRegister.DROPPED_PKT_CNT.cmd(0)) 

214 commands.append(SPIFRegister.IN_PERIPH_PKT_CNT.cmd(0)) 

215 

216 # Configure the creation of packets from fields to keys using the 

217 # "standard" input to SPIF (X | P | Y) and convert to (Y | X) 

218 commands.extend([ 

219 set_field_mask(self.__pipe, 0, self.__input_mask), 

220 set_field_shift(self.__pipe, 0, self.__input_shift), 

221 set_field_limit(self.__pipe, 0, self.n_atoms)]) 

222 

223 # These are unused but set them to be sure 

224 for i in range(1, N_FIELDS): 

225 commands.extend([ 

226 set_field_mask(self.__pipe, i, 0), 

227 set_field_shift(self.__pipe, i, 0), 

228 set_field_limit(self.__pipe, i, 0)]) 

229 

230 # Don't filter 

231 commands.extend([ 

232 set_filter_mask(self.__pipe, i, 0) for i in range(N_FILTERS) 

233 ]) 

234 commands.extend([ 

235 set_filter_value(self.__pipe, i, 1) for i in range(N_FILTERS) 

236 ]) 

237 

238 # Configure the output routing key 

239 commands.append(set_mapper_key(self.__pipe, self.__base_key)) 

240 

241 # Configure the links to send packets to the 8 FPGAs using the 

242 # lower bits; the input key is against the reversed links list because 

243 # the input numbering on SPIF is upside-down to the FPGA numbering, and 

244 # we put out the fixed key and mask in the FPGA link order, so we need 

245 # to match that with what SPIF will output. 

246 commands.extend( 

247 set_input_key(self.__pipe, i, self.__spif_key(f)) 

248 for i, f in enumerate(reversed(SPIF_INPUT_FPGA_LINKS))) 

249 commands.extend( 

250 set_input_mask(self.__pipe, i, self.__spif_mask) 

251 for i in range(len(SPIF_INPUT_FPGA_LINKS))) 

252 commands.extend( 

253 set_input_route(self.__pipe, i, i) 

254 for i in range(len(SPIF_INPUT_FPGA_LINKS))) 

255 

256 # Send the start signal 

257 commands.append(SpiNNFPGARegister.START.cmd()) 

258 

259 return commands 

260 

261 def __spif_key(self, fpga_link_id: int) -> int: 

262 fpga = self.__fpga_index(fpga_link_id) 

263 return self.__base_key + fpga 

264 

265 @property 

266 @overrides(AbstractSendMeMulticastCommandsVertex.pause_stop_commands) 

267 def pause_stop_commands(self) -> Iterable[MultiCastCommand]: 

268 # Send the stop signal 

269 return [SpiNNFPGARegister.STOP.cmd()]

271 @property 

272 @overrides(AbstractSendMeMulticastCommandsVertex.timed_commands) 

273 def timed_commands(self) -> List[MultiCastCommand]: 

274 return []

276 @overrides(PopulationApplicationVertex.get_atom_key_map) 

277 def get_atom_key_map( 

278 self, pre_vertex: MachineVertex, partition_id: str, 

279 routing_info: RoutingInfo) -> Iterable[Tuple[int, int]]: 

280 # Work out which machine vertex 

281 start = pre_vertex.vertex_slice.lo_atom 

282 key_and_mask = self.get_machine_fixed_key_and_mask( 

283 pre_vertex, partition_id) 

284 end = pre_vertex.vertex_slice.hi_atom + 1 

285 n_key = key_and_mask.key & self.INPUT_MASK 

286 start += n_key 

287 neuron_id = pre_vertex.vertex_slice.lo_atom + n_key 

288 for n in range(start, end, self.INPUT_MASK + 1): 

289 key = key_and_mask.key | n 

290 yield (neuron_id, key) 

291 neuron_id += self.INPUT_MASK + 1
