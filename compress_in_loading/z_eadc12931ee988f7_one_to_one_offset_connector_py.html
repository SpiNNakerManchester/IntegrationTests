39class OneToOneOffsetConnector( 

40 AbstractGenerateConnectorOnMachine, 

41 AbstractGenerateConnectorOnHost): 

42 """ 

43 A Connector that connects each pre-neuron to a post-neuron offset by a 

44 specific amount, positive or negative. If this goes beyond the start or 

45 end of the post neurons, it can optionally wrap around. Additional options 

46 include a group size, where the offset and wrap is applied repeatedly to 

47 subsets of neurons. 

48 

49 In the current implementation it is assumed that the pre- and 

50 post-populations have the same number of neurons, and that the number of 

51 neurons is divisible by the group size if specified. The offset must also 

52 be smaller than the group size or the number of neurons. 

53 """ 

54 

55 __slots__ = ("__n_neurons_per_group", "__offset", "__wrap") 

56 

57 def __init__(self, offset: int, wrap: bool, 

58 n_neurons_per_group: Optional[int] = None, 

59 safe: bool = True, verbose: bool = False, 

60 callback: None = None): 

61 """ 

62 :param offset: 

63 The offset to apply to the pre-neuron index to get the post neuron 

64 index. This can be positive or negative. 

65 :param wrap: 

66 Whether to wrap around the start or end of the post neurons if the 

67 post neuron id is out of range. 

68 :param n_neurons_per_group: 

69 The number of neurons in each group. 

70 Must be a positive integer divisor of source.size 

71 :param safe: 

72 If ``True``, check that weights and delays have valid values. 

73 If ``False``, this check is skipped. 

74 :param verbose: 

75 Whether to output extra information about the connectivity to a 

76 CSV file 

77 :param callback: 

78 if given, a callable that display a progress bar on the terminal. 

79 

80 .. note:: 

81 Not supported by sPyNNaker. 

82 """ 

83 super().__init__(safe, callback, verbose) 

84 self.__n_neurons_per_group = n_neurons_per_group 

85 self.__offset = offset 

86 self.__wrap = wrap 

87 

88 def __n_connections(self, synapse_info: SynapseInformation) -> int: 

89 if self.__wrap: 

90 # If there is a wrap, there will always be a next connection 

91 return synapse_info.n_pre_neurons 

92 

93 n_groups = 1 

94 if self.__n_neurons_per_group is not None: 

95 n_groups = synapse_info.n_pre_neurons // self.__n_neurons_per_group 

96 

97 # If there isn't a wrap, there are always offset less per group 

98 return synapse_info.n_pre_neurons - (n_groups * abs(self.__offset))

124 # At most each post neuron will be targeted by one pre-neuron

133 @overrides(AbstractGenerateConnectorOnHost.create_synaptic_block) 

134 def create_synaptic_block( 

135 self, post_slices: Sequence[Slice], post_vertex_slice: Slice, 

136 synapse_type: int, synapse_info: SynapseInformation) -> NDArray: 

137 group_size = self.__n_neurons_per_group 

138 if group_size is None: 

139 group_size = synapse_info.n_pre_neurons 

140 

141 # pylint: disable=protected-access 

142 post_lo, post_hi = synapse_info.pre_population._view_range 

143 pre_lo, pre_hi = synapse_info.pre_population._view_range 

144 post_start = max(post_vertex_slice.lo_atom, post_lo) 

145 post_end = min(post_vertex_slice.hi_atom + 1, post_hi + 1) 

146 post_group, post_value = divmod(post_start, group_size) 

147 

148 pre_start = pre_lo + (post_group * group_size) 

149 pre_end = min(pre_start + group_size, pre_hi) 

150 

151 pres = list() 

152 posts = list() 

153 for post in range(post_start, post_end): 

154 pre = post - self.__offset 

155 if pre < pre_start: 

156 if self.__wrap: 

157 pre += group_size 

158 else: 

159 continue 

160 if pre > pre_end: 

161 if self.__wrap: 

162 pre -= group_size 

163 else: 

164 continue 

165 pres.append(pre) 

166 posts.append(post) 

167 

168 post_value += 1 

169 if post_value == group_size: 

170 post_value = 0 

171 pre_start += group_size 

172 pre_end = min(pre_start + group_size, pre_hi) 

173 if pre_start >= pre_hi: 

174 break 

175 

176 block = numpy.zeros(len(pres), dtype=self.NUMPY_SYNAPSES_DTYPE) 

177 block["source"] = pres 

178 block["target"] = posts 

179 block["weight"] = self._generate_weights( 

180 block["source"], block["target"], len(pres), post_vertex_slice, 

181 synapse_info) 

182 block["delay"] = self._generate_delays( 

183 block["source"], block["target"], len(pres), post_vertex_slice, 

184 synapse_info) 

185 block["synapse_type"] = synapse_type 

186 return block

213 @overrides(AbstractConnector.validate_connection) 

214 def validate_connection( 

215 self, application_edge: ProjectionApplicationEdge, 

216 synapse_info: SynapseInformation) -> None: 

217 if (synapse_info.pre_population.size != 

218 synapse_info.post_population.size): 

219 raise NotImplementedError( 

220 "OneToOneOffsetConnector is only designed to be used with " 

221 "populations that are the same size as each other") 

222 if self.__n_neurons_per_group is not None: 

223 if self.__n_neurons_per_group > synapse_info.pre_population.size: 

224 raise ValueError( 

225 "OneToOneOffsetConnector cannot be used with a group size " 

226 "larger than the population size") 

227 if ((synapse_info.post_population.size / 

228 self.__n_neurons_per_group) != 

229 (synapse_info.post_population.size // 

230 self.__n_neurons_per_group)): 

231 raise NotImplementedError( 

232 "The number of neurons in each population must be " 

233 "divisible by the number of neurons per group") 

234 

235 n_values = self.__n_neurons_per_group 

236 if n_values is None: 

237 n_values = synapse_info.n_pre_neurons 

238 if n_values < abs(self.__offset): 

239 raise ValueError( 

240 "The offset must be smaller than the number of neurons")
