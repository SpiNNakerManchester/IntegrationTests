139 def _write_common_data_spec( 

140 self, spec: DataSpecificationGenerator, 

141 rec_regions: Sequence[int]) -> None: 

142 """ 

143 Write the data specification for the common regions. 

144 

145 :param spec: 

146 The data specification to write to 

147 :param rec_regions: 

148 A list of sizes of each recording region (including empty ones) 

149 """ 

150 # Write the setup region 

151 spec.reserve_memory_region( 

152 region=self.__regions.system, size=SIMULATION_N_BYTES, 

153 label='System') 

154 spec.switch_write_focus(self.__regions.system) 

155 spec.write_array(get_simulation_header_array(self.__binary_file_name)) 

156 

157 # Reserve memory for provenance 

158 self.reserve_provenance_data_region(spec) 

159 

160 # Write profile data 

161 reserve_profile_region( 

162 spec, self.__regions.profile, self._pop_vertex.n_profile_samples) 

163 write_profile_region_data( 

164 spec, self.__regions.profile, self._pop_vertex.n_profile_samples) 

165 

166 # Set up for recording 

167 spec.reserve_memory_region( 

168 region=self.__regions.recording, 

169 size=get_recording_header_size(len(rec_regions)), 

170 label="Recording") 

171 spec.switch_write_focus(self.__regions.recording) 

172 spec.write_array(get_recording_header_array(rec_regions))
