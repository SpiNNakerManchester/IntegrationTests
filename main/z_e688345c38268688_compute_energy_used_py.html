51def compute_energy_used(checkpoint: Optional[int] = None) -> PowerUsed: 

52 """ 

53 This algorithm does the actual work of computing energy used by a 

54 simulation (or other application) running on SpiNNaker. 

55 

56 :param checkpoint: the time at which to compute execution energy up to 

57 :returns: Summary object of power used 

58 """ 

59 # Get data from provenance 

60 with GlobalProvenance() as db: 

61 waiting_ms = db.get_category_timer_sum_by_reset(TimerCategory.WAITING) 

62 setup_ms = db.get_timer_sum_by_category_and_reset( 

63 TimerCategory.SETTING_UP) 

64 get_machine_ms = db.get_timer_sum_by_category_and_reset( 

65 TimerCategory.GET_MACHINE) 

66 

67 mapping_ms = db.get_timer_sum_by_category_and_reset( 

68 TimerCategory.MAPPING) 

69 loading_ms = db.get_timer_sum_by_category_and_reset( 

70 TimerCategory.LOADING) 

71 

72 run_other_ms = db.get_timer_sum_by_category_and_reset( 

73 TimerCategory.RUN_OTHER) 

74 run_loop_ms = db.get_timer_sum_by_category_and_reset( 

75 TimerCategory.RUN_LOOP) 

76 resetting_ms = db.get_timer_sum_by_category_and_reset( 

77 TimerCategory.RESETTING) 

78 

79 shutting_down_ms = db.get_timer_sum_by_category_and_reset( 

80 TimerCategory.SHUTTING_DOWN) 

81 

82 # Separate out processes that are part of the others but that happen 

83 # on the machine, so we can account for active machine, not idle 

84 data_loading_ms = db.get_timer_sum_by_work(TimerWork.LOADING_DATA) 

85 loading_ms -= data_loading_ms 

86 data_extraction_ms = 0 

87 if get_config_bool("Machine", "enable_advanced_monitor_support"): 

88 data_extraction_ms = db.get_timer_sum_by_work( 

89 TimerWork.EXTRACT_DATA) 

90 run_loop_ms -= data_extraction_ms 

91 expansion_ms = db.get_timer_sum_by_work(TimerWork.SYNAPSE) 

92 loading_ms -= expansion_ms 

93 

94 if checkpoint is not None: 

95 execute_on_machine_ms = checkpoint 

96 else: 

97 timesteps = FecDataView.get_current_run_timesteps() 

98 if timesteps is None: 

99 raise ValueError( 

100 "Cannot compute energy without knowing the end time") 

101 ts_factor = FecDataView.get_time_scale_factor() 

102 execute_on_machine_ms = int(round(timesteps * ts_factor)) 

103 

104 run_loop_ms -= execute_on_machine_ms 

105 

106 machine = FecDataView.get_machine() 

107 version = FecDataView.get_machine_version() 

108 n_boards = len(machine.ethernet_connected_chips) 

109 n_chips = machine.n_chips 

110 n_cores = FecDataView.get_n_placements() 

111 n_frames = _calculate_n_frames(machine) 

112 

113 active_cores: Dict[Tuple[int, int], int] = defaultdict(int) 

114 power_cores: Dict[Tuple[int, int], int] = {} 

115 n_active_cores = 0 

116 for pl in FecDataView.iterate_placemements(): 

117 if not isinstance(pl.vertex, AbstractHasAssociatedBinary): 

118 continue 

119 vertex: AbstractHasAssociatedBinary = cast( 

120 AbstractHasAssociatedBinary, pl.vertex) 

121 if vertex.get_binary_start_type() != ExecutableType.SYSTEM: 

122 if isinstance(vertex, ChipPowerMonitorMachineVertex): 

123 power_cores[(pl.x, pl.y)] = pl.p 

124 else: 

125 active_cores[(pl.x, pl.y)] += 1 

126 n_active_cores += 1 

127 n_active_chips = len(active_cores) 

128 

129 # TODO confirm Power monitors are not included here 

130 extra_monitors_per_chip = (version.n_scamp_cores 

131 + FecDataView.get_all_monitor_cores() - 1) 

132 extra_monitors_per_board = (version.n_scamp_cores + 

133 FecDataView.get_ethernet_monitor_cores() - 1) 

134 if get_config_bool("Reports", "write_energy_report"): 

135 run_chip_active_time = _extract_cores_active_time( 

136 checkpoint, active_cores, power_cores, version) 

137 else: 

138 run_chip_active_time = _assume_core_always_active( 

139 active_cores, execute_on_machine_ms) 

140 load_chip_active_time = _make_extra_monitor_core_use( 

141 data_loading_ms, machine, extra_monitors_per_board, 

142 extra_monitors_per_chip) 

143 extraction_chip_active_time = _make_extra_monitor_core_use( 

144 data_extraction_ms, machine, extra_monitors_per_board, 

145 extra_monitors_per_chip) 

146 

147 run_router_packets = _extract_router_packets("Run", version) 

148 load_router_packets = _extract_router_packets("Load", version) 

149 extraction_router_packets = _extract_router_packets("Extract", version) 

150 

151 # TODO get_machine not include here 

152 return compute_energy_over_time( 

153 waiting_ms, setup_ms, get_machine_ms, mapping_ms, loading_ms, 

154 data_loading_ms, expansion_ms, data_extraction_ms, run_other_ms, 

155 run_loop_ms, execute_on_machine_ms, resetting_ms, shutting_down_ms, 

156 version, n_chips, n_active_chips, n_boards, n_frames, n_cores, 

157 n_active_cores, load_chip_active_time, extraction_chip_active_time, 

158 run_chip_active_time, load_router_packets, extraction_router_packets, 

159 run_router_packets)

162def _calculate_n_frames(machine: Machine) -> int: 

163 """ 

164 Figures out how many frames are being used in this setup. 

165 A key of cabinet,frame will be used to identify unique frame. 

166 

167 :param machine: the machine object 

168 :return: number of frames 

169 """ 

170 # if not spalloc, then could be any type of board, but unknown cooling 

171 if not FecDataView.has_allocation_controller(): 

172 return 0 

173 

174 # using spalloc in some form; how many unique frames? 

175 cabinet_frame = set() 

176 mac = FecDataView.get_allocation_controller() 

177 for ethernet_connected_chip in machine.ethernet_connected_chips: 

178 cabinet, frame, _ = mac.where_is_machine( 

179 ethernet_connected_chip.x, ethernet_connected_chip.y) 

180 cabinet_frame.add((cabinet, frame)) 

181 return len(cabinet_frame)

184def _extract_router_packets( 

185 prefix: str, version: AbstractVersion) -> RouterPackets: 

186 packets_per_chip: RouterPackets = defaultdict(dict) 

187 with ProvenanceReader() as db: 

188 for name in version.get_router_report_packet_types(): 

189 for (x, y, value) in db.get_router_by_chip(f"{prefix}{name}"): 

190 packets_per_chip[x, y][name] = value 

191 return packets_per_chip

194def _extract_cores_active_time( 

195 checkpoint: Optional[int], active_cores: Dict[Tuple[int, int], int], 

196 power_cores: Dict[Tuple[int, int], int], 

197 version: AbstractVersion) -> ChipActiveTime: 

198 sampling_frequency = get_config_int("EnergyMonitor", "sampling_frequency") 

199 

200 chip_activity: ChipActiveTime = {} 

201 with BufferDatabase() as buff_db: 

202 for (x, y), n_cores in active_cores.items(): 

203 # Find the core that was used on this chip for power monitoring 

204 p = power_cores[(x, y)] 

205 # Get time per sample in seconds (frequency in microseconds) 

206 time_for_recorded_sample_s = sampling_frequency / _US_PER_SECOND 

207 data, _missing = buff_db.get_recording(x, y, p, RECORDING_CHANNEL) 

208 results = numpy.frombuffer(data, dtype=numpy.uint32).reshape( 

209 -1, version.max_cores_per_chip + 1) 

210 # Get record times in milliseconds (frequency in microseconds) 

211 record_times = results[:, 0] * sampling_frequency / _US_PER_MS 

212 # The remaining columns are the counts of active / inactive at 

213 # each sample point 

214 activity = results[:, 1:].astype(numpy.float64) 

215 # Set the activity of *this* core to 0, as we don't want to 

216 # measure that! 

217 physical_core = FecDataView.get_physical_core_id((x, y), p) 

218 activity[:, physical_core] = 0 

219 # Convert to actual active time, assuming the core is fully active 

220 # or fully inactive between samples 

221 activity_times = activity * time_for_recorded_sample_s 

222 # If checkpoint is specified, filter the times 

223 if checkpoint is not None: 

224 activity_times = activity_times[record_times < checkpoint] 

225 chip_activity[x, y] = (activity_times.sum(), n_cores) 

226 return chip_activity

229def _assume_core_always_active( 

230 active_cores: Dict[Tuple[int, int], int], 

231 execute_on_machine_ms: float) -> ChipActiveTime: 

232 """ 

233 As there are no power monitors assume cores always active 

234 

235 """ 

236 logger.warning( 

237 "Energy monitoring cores not enabled, assuming all cores were" 

238 " active for whole run time. To get a better energy estimate," 

239 " set write_energy_report=True in the [Reports] section of the" 

240 " configuration file") 

241 chip_activity: ChipActiveTime = {} 

242 for (x, y), n_cores in active_cores.items(): 

243 chip_activity[x, y] = ( 

244 (execute_on_machine_ms * n_cores) / _MS_PER_SECOND, n_cores) 

245 return chip_activity

248def _make_extra_monitor_core_use( 

249 time_ms: int, machine: Machine, extra_monitors_per_board: int, 

250 extra_monitors_per_chip: int) -> ChipActiveTime: 

251 time_s = time_ms / _MS_PER_SECOND 

252 core_use = {} 

253 for chip in machine.chips: 

254 n_monitors = extra_monitors_per_chip 

255 if chip.ip_address is not None: 

256 n_monitors += extra_monitors_per_board 

257 core_use[chip.x, chip.y] = (n_monitors * time_s, n_monitors) 

258 return core_use

261def compute_energy_over_time( 

262 waiting_ms: float, setup_ms: float, get_machine_ms: float, 

263 mapping_ms: float, loading_ms: float, data_loading_ms: float, 

264 expansion_ms: float, data_extraction_ms: float, run_other_ms: float, 

265 run_loop_ms: float, execute_on_machine_ms: float, 

266 resetting_ms: float, shutting_down_ms: float, version: AbstractVersion, 

267 n_chips: int, n_active_chips: int, n_boards: int, n_frames: int, 

268 n_cores: int, n_active_cores: int, 

269 load_chip_active_time: ChipActiveTime, 

270 extraction_chip_active_time: ChipActiveTime, 

271 run_chip_active_time: ChipActiveTime, 

272 load_router_packets: RouterPackets, 

273 extraction_router_packets: RouterPackets, 

274 run_router_packets: RouterPackets) -> PowerUsed: 

275 """ 

276 Compute the energy used by a simulation running on SpiNNaker. 

277 

278 :param waiting_ms: time spent waiting for things to happen in general 

279 :param setup_ms: time spent setting up the simulation 

280 :param get_machine_ms: time spent getting the machine 

281 :param mapping_ms: time spent mapping to the machine 

282 :param loading_ms: 

283 time spent loading the simulation onto the passive machine 

284 :param data_loading_ms: 

285 time spent loading data onto the machine actively using the machine to 

286 load the data 

287 :param expansion_ms: time spent expanding the data on the machine 

288 :param data_extraction_ms: 

289 time spent extracting data from the machine actively using the machine 

290 to extract the data 

291 :param run_other_ms: 

292 time spent in running but not the active machine time, just in between 

293 calls to other things 

294 :param run_loop_ms: 

295 time spent in running but not the active machine time, just in the 

296 run loop itself 

297 :param execute_on_machine_ms: 

298 time spent executing the simulation on the machine actively using it 

299 :param resetting_ms: time spent resetting the simulation 

300 :param shutting_down_ms: time spent shutting down the simulation 

301 :param version: the version of the machine 

302 :param n_chips: number of chips that make up the machine 

303 :param n_active_chips: number of chips active in simulation 

304 :param n_boards: number of boards that make up the machine 

305 :param n_frames: number of frames that make up the machine 

306 :param n_cores: number of cores that are used by the simulation 

307 :param n_active_cores: number of cores actively used by the simulation 

308 :param load_chip_active_time: 

309 time that each core was active during loading 

310 :param extraction_chip_active_time: 

311 time that each core was active during extraction 

312 :param run_chip_active_time: 

313 time that each core was active during running 

314 :param load_router_packets: 

315 packets sent by the machine during loading 

316 :param extraction_router_packets: 

317 packets sent by the machine during extraction 

318 :param run_router_packets: 

319 packets sent by the machine during running 

320 :returns: Summary object of power used 

321 """ 

322 

323 # Time and energy spent on the host machine, with the machine (at least 

324 # theoretically) running, doing general software tasks that we don't want 

325 # to put in other categories. 

326 other_time_s = ( 

327 waiting_ms + setup_ms + get_machine_ms + shutting_down_ms + 

328 run_other_ms + run_loop_ms + resetting_ms) / _MS_PER_SECOND 

329 other_energy_j = version.get_idle_energy( 

330 other_time_s, n_frames, n_boards, n_chips) 

331 

332 # Time and energy mapping to the machine 

333 mapping_time_s = mapping_ms / _MS_PER_SECOND 

334 mapping_energy_j = version.get_idle_energy( 

335 mapping_time_s, n_frames, n_boards, n_chips) 

336 

337 # Time and energy spent loading data onto the machine 

338 loading_time_s = ( 

339 loading_ms + data_loading_ms + expansion_ms) / _MS_PER_SECOND 

340 loading_energy_j = version.get_idle_energy( 

341 loading_ms / _MS_PER_SECOND, n_frames, n_boards, n_chips) 

342 loading_energy_j += version.get_active_energy( 

343 (data_loading_ms + expansion_ms) / _MS_PER_SECOND, n_frames, n_boards, 

344 n_chips, load_chip_active_time, load_router_packets) 

345 

346 # Time and energy spent extracting data from the machine 

347 saving_time_s = data_extraction_ms / _MS_PER_SECOND 

348 saving_energy_j = version.get_active_energy( 

349 saving_time_s, n_frames, n_boards, n_chips, 

350 extraction_chip_active_time, extraction_router_packets) 

351 

352 # Time and energy spent running the simulation on the machine 

353 exec_time_s = execute_on_machine_ms / _MS_PER_SECOND 

354 exec_energy_j = version.get_active_energy( 

355 exec_time_s, n_frames, n_boards, n_chips, 

356 run_chip_active_time, run_router_packets) 

357 exec_energy_cores_j = version.get_active_energy( 

358 exec_time_s, 0, 0, n_active_chips, run_chip_active_time, 

359 run_router_packets) 

360 exec_energy_boards_j = version.get_active_energy( 

361 exec_time_s, 0, n_boards, n_chips, run_chip_active_time, 

362 run_router_packets) 

363 

364 return PowerUsed( 

365 n_chips, n_active_chips, n_cores, n_active_cores, n_boards, n_frames, 

366 exec_time_s, mapping_time_s, loading_time_s, saving_time_s, 

367 other_time_s, exec_energy_j, exec_energy_cores_j, exec_energy_boards_j, 

368 mapping_energy_j, loading_energy_j, saving_energy_j, other_energy_j)
