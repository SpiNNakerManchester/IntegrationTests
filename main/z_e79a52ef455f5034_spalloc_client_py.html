85def fix_url(url: Any) -> str: 

86 """ 

87 Makes sure the url is the correct format. 

88 

89 :param url: original url 

90 :returns: cleaned url 

91 """ 

92 parts = urlparse(url) 

93 if parts.scheme != 'https': 

94 parts = ParseResult("https", parts.netloc, parts.path, 

95 parts.params, parts. query, parts.fragment) 

96 if not parts.path.endswith("/"): 

97 parts = ParseResult(parts.scheme, parts.netloc, parts.path + "/", 

98 parts.params, parts.query, parts.fragment) 

99 return urlunparse(parts)

102def get_n_boards() -> int: 

103 """ 

104 Works out how many boards are needed. 

105 

106 :return: Number of boards needed with a safety factor 

107 :raises ~spinn_utilities.exceptions.SpiNNUtilsException: 

108 If data needed is not available 

109 """ 

110 if SpiNNManDataView.has_n_boards_required(): 

111 return SpiNNManDataView.get_n_boards_required() 

112 else: 

113 n_chips = SpiNNManDataView.get_n_chips_needed() 

114 # reduce max chips by 2 in case you get a bad board(s) 

115 chips_div = ( 

116 SpiNNManDataView.get_machine_version().n_chips_per_board - 2) 

117 n_boards_float = float(n_chips) / chips_div 

118 logger.info("{:.2f} Boards Required for {} chips", 

119 n_boards_float, n_chips) 

120 # If the number of boards rounded up is less than 50% of a board 

121 # bigger than the actual number of boards, 

122 # add another board just in case. 

123 n_boards = int(math.ceil(n_boards_float)) 

124 if n_boards - n_boards_float < 0.5: 

125 n_boards += 1 

126 return n_boards

438class _ProxyReceiver(threading.Thread): 

439 """ 

440 Receives all messages off an open websocket and dispatches them to 

441 registered listeners. 

442 """ 

443 

444 def __init__(self, websocket: WebSocket): 

445 """ 

446 :param websocket: WebSocket obtained when starting the client 

447 """ 

448 super().__init__(daemon=True) 

449 self.__ws = websocket 

450 self.__returns: Dict[int, _WSCB] = {} 

451 self.__handlers: Dict[int, _WSCB] = {} 

452 self.__correlation_id = 0 

453 self.__closed = False 

454 self.start() 

455 

456 def run(self) -> None: 

457 """ 

458 The handler loop of this thread. 

459 """ 

460 while self.__ws.connected: 

461 try: 

462 result: Tuple[int, bytes] = self.__ws.recv_data() 

463 frame = result[1] 

464 if len(frame) < _msg.size: 

465 # Message is out of protocol 

466 continue 

467 except Exception: # pylint: disable=broad-except 

468 # If closed, ignore error and get out of here 

469 if self.__closed: 

470 break 

471 

472 # Make someone aware of the error 

473 logger.exception("Error in websocket before close") 

474 

475 # If we are disconnected before closing, make errors happen 

476 if not self.__ws.connected: 

477 for rt in self.__returns.values(): 

478 rt(None) 

479 for hd in self.__handlers.values(): 

480 hd(None) 

481 break 

482 code, num = _msg.unpack_from(frame, 0) 

483 if code == ProxyProtocol.MSG: 

484 self.dispatch_message(num, frame) 

485 else: 

486 self.dispatch_return(num, frame) 

487 

488 def expect_return(self, handler: _WSCB) -> int: 

489 """ 

490 Register a one-shot listener for a call-like message's return. 

491 

492 :return: The correlation ID 

493 """ 

494 c = self.__correlation_id 

495 self.__correlation_id += 1 

496 self.__returns[c] = handler 

497 return c 

498 

499 def listen(self, channel_id: int, handler: _WSCB) -> None: 

500 """ 

501 Register a persistent listener for one-way messages. 

502 """ 

503 self.__handlers[channel_id] = handler 

504 

505 def dispatch_return(self, correlation_id: int, msg: bytes) -> None: 

506 """ 

507 Dispatch a received call-return message. 

508 """ 

509 handler = self.__returns.pop(correlation_id, None) 

510 if handler: 

511 handler(msg) 

512 

513 def dispatch_message(self, channel_id: int, msg: bytes) -> None: 

514 """ 

515 Dispatch a received one-way message. 

516 """ 

517 handler = self.__handlers.get(channel_id) 

518 if handler: 

519 handler(msg) 

520 

521 def unlisten(self, channel_id: int) -> None: 

522 """ 

523 Deregister a listener for a channel 

524 """ 

525 self.__handlers.pop(channel_id) 

526 

527 def close(self) -> None: 

528 """ 

529 Mark receiver closed to avoid errors 

530 """ 

531 self.__closed = True

680 @overrides(SpallocJob.wait_until_ready)

683 retries = 0 

684 

685 if self.__board_st is None: 

686 queue_time = get_config_int_or_none( 

687 "Machine", "spalloc_queue_time") 

688 if queue_time is None: 

689 n_retries = sys.maxsize 

690 else: 

691 n_retries = queue_time // 5 

692 else: 

693 n_retries = 3 

694 

695 while state == SpallocState.QUEUED: 

696 logger.info(f"Waiting as job is QUEUED {retries=} of {n_retries}") 

697 if retries >= n_retries: 

698 if self.__board_st is None: 

699 raise SpallocBoardUnavailableException( 

700 f"{self._url} killed " 

701 f"as it remained QUEUED for {queue_time} seconds") 

702 else: 

703 raise SpallocBoardUnavailableException( 

704 f"Boards described as {self.__board_st} " 

705 f"are not available") 

706 time.sleep(5) 

707 retries += 1 

708 state = self.get_state() 

709 

710 while state == SpallocState.POWER: 

711 logger.info(f"Waiting as job is powering up {retries=}") 

712 time.sleep(5) 

713 retries += 1 

714 state = self.get_state() 

715 

716 if state != SpallocState.READY: 

717 raise SpallocException(f"job was unexpectedly {state=}")

802class _ProxiedConnection(metaclass=AbstractBase): 

803 """ 

804 Core multiplexer/demultiplexer emulating a connection that is proxied 

805 over a websocket. 

806 

807 None of the methods are public because subclasses may expose a profile of 

808 them to conform to a particular type of connection. 

809 """ 

810 

811 def __init__(self, websocket: WebSocket, receiver: _ProxyReceiver): 

812 """ 

813 :param websocket: WebSocket obtained when starting the client 

814 :param receiver: Receiver created when starting the Client 

815 """ 

816 self.__ws: Optional[WebSocket] = websocket 

817 self.__receiver: Optional[_ProxyReceiver] = receiver 

818 self.__msgs: queue.SimpleQueue = queue.SimpleQueue() 

819 self.__call_queue: queue.Queue = queue.Queue(1) 

820 self.__call_lock = threading.RLock() 

821 self.__current_msg: Optional[bytes] = None 

822 self.__handle = self._open_connection() 

823 self.__receiver.listen(self.__handle, self.__msgs.put) 

824 

825 @abstractmethod 

826 def _open_connection(self) -> int: 

827 raise NotImplementedError 

828 

829 def _call(self, protocol: ProxyProtocol, packer: struct.Struct, 

830 unpacker: struct.Struct, *args: int) -> Tuple[Any, ...]: 

831 """ 

832 Do a synchronous call. 

833 

834 :param protocol: 

835 The protocol message number. 

836 :param packer: 

837 How to form the protocol message. The first two arguments passed 

838 will be the protocol message number and an issued correlation ID 

839 (not needed by the caller). 

840 :param unpacker: 

841 How to extract the expected response. 

842 :param args: 

843 Additional arguments to pass to the packer. 

844 :return: 

845 The results from the unpacker *after* the protocol message code and 

846 the correlation ID. 

847 :raises IOError: 

848 If something goes wrong. This could be due to the websocket being 

849 closed, or the receipt of an ERROR response. 

850 """ 

851 if not self._connected: 

852 raise IOError("socket closed") 

853 if not self.__receiver: 

854 raise IOError("socket closed") 

855 if not self.__ws: 

856 raise IOError("socket closed") 

857 with self.__call_lock: 

858 # All calls via websocket use correlation_id 

859 correlation_id = self.__receiver.expect_return( 

860 self.__call_queue.put) 

861 self.__ws.send_binary(packer.pack(protocol, correlation_id, *args)) 

862 if not self._connected: 

863 raise IOError("socket closed after send!") 

864 reply = self.__call_queue.get() 

865 code, _ = _msg.unpack_from(reply, 0) 

866 if code == ProxyProtocol.ERROR: 

867 # Rest of message is UTF-8 encoded error message string 

868 payload = reply[_msg.size:].decode("utf-8") 

869 if len(payload): 

870 raise _ProxyServiceError(payload) 

871 raise _ProxyServiceError( 

872 f"unknown problem with {protocol} call") 

873 return unpacker.unpack(reply)[2:] 

874 

875 @property 

876 def _connected(self) -> bool: 

877 return bool(self.__ws and self.__ws.connected) 

878 

879 def _throw_if_closed(self) -> None: 

880 if not self._connected: 

881 raise IOError("socket closed") 

882 

883 def _close(self) -> None: 

884 if self._connected: 

885 channel_id, = self._call( 

886 ProxyProtocol.CLOSE, _close_req, _open_close_res, 

887 self.__handle) 

888 if channel_id != self.__handle: 

889 raise IOError("failed to close proxy socket") 

890 if self.__receiver: 

891 self.__receiver.unlisten(self.__handle) 

892 self.__ws = None 

893 self.__receiver = None 

894 

895 def _send(self, message: bytes) -> None: 

896 self._throw_if_closed() 

897 # Put the header on the front and send it 

898 if not self.__ws: 

899 raise IOError("socket closed") 

900 self.__ws.send_binary(_msg.pack( 

901 ProxyProtocol.MSG, self.__handle) + message) 

902 

903 def _send_to(self, message: bytes, x: int, y: int, port: int) -> None: 

904 self._throw_if_closed() 

905 # Put the header on the front and send it 

906 if not self.__ws: 

907 raise IOError("socket closed") 

908 self.__ws.send_binary(_msg_to.pack( 

909 ProxyProtocol.MSG_TO, self.__handle, x, y, port) + message) 

910 

911 def __get(self, timeout: float = 0.5) -> bytes: 

912 """ 

913 Get a value from the queue. Handles block/non-block switching and 

914 trimming of the message protocol prefix. 

915 """ 

916 if not timeout: 

917 return self.__msgs.get(block=False)[_msg.size:] 

918 else: 

919 return self.__msgs.get(timeout=timeout)[_msg.size:] 

920 

921 def _receive(self, timeout: Optional[float] = None) -> bytes: 

922 if self.__current_msg is not None: 

923 try: 

924 return self.__current_msg 

925 finally: 

926 self.__current_msg = None 

927 if timeout is None: 

928 while True: 

929 try: 

930 return self.__get() 

931 except queue.Empty: 

932 self._throw_if_closed() 

933 else: 

934 try: 

935 return self.__get(timeout) 

936 except queue.Empty as e: 

937 self._throw_if_closed() 

938 raise SpinnmanTimeoutException("receive", timeout) from e 

939 

940 def _is_ready_to_receive(self, timeout: float = 0) -> bool: 

941 # If we already have a message or the queue peek succeeds, return now 

942 if self.__current_msg is not None or not self.__msgs.empty(): 

943 return True 

944 try: 

945 self.__current_msg = self.__get(timeout) 

946 return True 

947 except queue.Empty: 

948 return False
