136 @overrides(AbstractPlasticSynapseDynamics.get_plastic_synaptic_data) 

137 def get_plastic_synaptic_data( 

138 self, connections: ConnectionsArray, 

139 connection_row_indices: NDArray[integer], n_rows: int, 

140 n_synapse_types: int, 

141 max_n_synapses: int, max_atoms_per_core: int) -> Tuple[ 

142 NDArray[uint32], NDArray[uint32], NDArray[uint32], 

143 NDArray[uint32]]: 

144 weights = numpy.rint(numpy.abs(connections["weight"])) 

145 n_neuron_id_bits = get_n_bits(max_atoms_per_core) 

146 neuron_id_mask = (1 << n_neuron_id_bits) - 1 

147 fixed_plastic = ( 

148 ((weights.astype(uint32) & 0xFFFF) << 16) | 

149 (connections["synapse_type"].astype(uint32) << n_neuron_id_bits) | 

150 (connections["target"] & neuron_id_mask)) 

151 fixed_plastic_rows = self.convert_per_connection_data_to_rows( 

152 connection_row_indices, n_rows, 

153 fixed_plastic.view(dtype=uint8).reshape((-1, BYTES_PER_WORD)), 

154 max_n_synapses) 

155 

156 flags = 0x80000000 | numpy.arange(n_rows, dtype=uint32) 

157 

158 fp_size = self.get_n_items(fixed_plastic_rows, BYTES_PER_WORD) 

159 fp_data = numpy.vstack([ 

160 fixed_row.view(uint32) for fixed_row in fixed_plastic_rows]) 

161 pp_data = flags.astype(uint32) 

162 pp_size = numpy.ones(n_rows, dtype=uint32) 

163 

164 return fp_data, pp_data, fp_size, pp_size

188 @overrides(AbstractPlasticSynapseDynamics.read_plastic_synaptic_data) 

189 def read_plastic_synaptic_data( 

190 self, n_synapse_types: int, 

191 pp_size: NDArray[uint32], pp_data: List[NDArray[uint32]], 

192 fp_size: NDArray[uint32], fp_data: List[NDArray[uint32]], 

193 max_atoms_per_core: int) -> ConnectionsArray: 

194 data = numpy.concatenate(fp_data) 

195 weight = ((data >> 16) & 0xFFFF).astype(int16) 

196 n_neuron_id_bits = get_n_bits(max_atoms_per_core) 

197 neuron_id_mask = (1 << n_neuron_id_bits) - 1 

198 connections = numpy.zeros(data.size, dtype=NUMPY_CONNECTORS_DTYPE) 

199 connections["source"] = numpy.concatenate( 

200 [numpy.repeat(i, fp_size[i]) for i in range(len(fp_size))]) 

201 connections["target"] = data & neuron_id_mask 

202 connections["weight"] = weight 

203 connections["delay"] = 1 

204 return connections
