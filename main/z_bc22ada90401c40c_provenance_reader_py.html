27class ProvenanceReader(BaseDatabase): 

28 """ 

29 Provides a connection to a database containing provenance for the current 

30 run and some convenience methods for extracting provenance data from it. 

31 

32 By default this will wrap around the database used to store the 

33 provenance of the last run. The path is not updated so this reader is 

34 not effected by a reset or an end. 

35 

36 The assumption is that the database is in the current provenance format. 

37 This includes both that DDL statements used to create the database but 

38 also the underlying database structure (currently sqlite3) 

39 

40 .. warning:: 

41 This class is only a wrapper around the database file so if the file 

42 is deleted the class will no longer work. 

43 """ 

44 

45 __slots__ = ()

50

52 ``provenance_data_path`` parameter, allowing 

53 :py:meth:`get_last_run_database_path` to find the correct path. 

54 

55 :param provenance_data_path: Path to the provenance database to wrap 

56 """

58 row_factory=None, text_factory=None)

60 def run_query(self, query: str, params: Iterable[_SqliteTypes] = () 

61 ) -> List[Sequence[_SqliteTypes]]: 

62 """ 

63 Opens a connection to the database, runs a query, extracts the results 

64 and closes the connection. 

65 

66 The return type depends on the use_sqlite_rows parameter. 

67 By default this method returns tuples (lookup by index) but the 

68 advanced tuple type can be used instead, which supports lookup by name 

69 used in the query (use ``AS name`` in the query to set). 

70 

71 This method will not allow queries that change the database unless the 

72 read_only flag is set to False. 

73 

74 .. note:: 

75 This method is mainly provided as a support method for the later 

76 methods that return specific data. For new IntergationTests 

77 please add a specific method rather than call this directly. 

78 

79 :param query: The SQL query to be run. May include ``?`` wildcards 

80 :param params: 

81 The values to replace the ``?`` wildcards with. 

82 The number and types must match what the query expects 

83 :return: A list possibly empty of tuples/rows 

84 (one for each row in the database) 

85 where the number and type of the values corresponds to the where 

86 statement 

87 """ 

88 return list(self.cursor().execute(query, list(params)))

90 def cores_with_late_spikes(self) -> List[Tuple[int, int, int, int]]: 

91 """ 

92 Gets the x, y, p and count of the cores where late spikes arrived. 

93 

94 Cores that received spikes but where none were late are *not* included. 

95 

96 :return: A list hopefully empty of tuples (x, y, p, count) of cores 

97 where their where late arriving spikes. 

98 """ 

99 query = """ 

100 SELECT x, y, p, the_value AS "value" 

101 FROM core_provenance 

102 WHERE description = 'Number_of_late_spikes' 

103 AND the_value > 0 

104 """ 

105 return cast(List[Tuple[int, int, int, int]], self.run_query(query))

107 def get_provenance_for_router(self, x: int, y: int) -> str: 

108 """ 

109 Gets the provenance item(s) from the last run relating to a chip. 

110 

111 :param x: The X coordinate of the chip 

112 :param y: The Y coordinate of the chip 

113 :return: 

114 A possibly multi-line string with for each row which matches the 

115 like a line ``description_name: value`` 

116 """ 

117 query = """ 

118 SELECT 

119 description, 

120 sum(the_value) AS "value" 

121 FROM router_provenance 

122 WHERE x = ? AND y = ? 

123 GROUP BY description 

124 ORDER BY description 

125 """ 

126 return "\n".join( 

127 f"{cast(str, row[0])}: {cast(int, row[1])}" 

128 for row in self.run_query(query, [int(x), int(y)]))

131 """ 

132 Gets the cores with provenance. 

133 

134 :return: A list tuples (x, y, p) 

135 """ 

136 query = """ 

137 SELECT core_name, x, y, p 

138 FROM core_provenance_view

141 return cast(List[XYP], self.run_query(query))

180 """
