44def insert_extra_monitor_vertices_to_graphs(placements: Placements) -> Tuple[ 

45 Dict[Chip, DataSpeedUpPacketGatherMachineVertex], 

46 Dict[Chip, ExtraMonitorSupportMachineVertex]]: 

47 """ 

48 Inserts the extra monitor vertices into the graph that correspond to 

49 the extra monitor cores required. 

50 

51 :param placements: 

52 :return: mapping from *Ethernet-enabled* chip locations to their gatherer, 

53 mapping from *all* chip locations to their extra monitor 

54 """ 

55 chip_to_gatherer_map = dict() 

56 chip_to_monitor_map = dict() 

57 machine = FecDataView.get_machine() 

58 ethernet_chips = machine.ethernet_connected_chips 

59 progress = ProgressBar( 

60 len(ethernet_chips), "Inserting extra monitors into graphs") 

61 

62 for eth in progress.over(ethernet_chips): 

63 assert eth.ip_address is not None 

64 gatherer = DataSpeedUpPacketGatherMachineVertex( 

65 x=eth.x, y=eth.y, ip_address=eth.ip_address) 

66 chip_to_gatherer_map[eth] = gatherer 

67 p = pick_core_for_system_placement(placements, eth) 

68 placements.add_placement(Placement(gatherer, eth.x, eth.y, p)) 

69 for chip in machine.get_chips_by_ethernet(eth.x, eth.y): 

70 monitor = ExtraMonitorSupportMachineVertex() 

71 chip_to_monitor_map[chip] = monitor 

72 p = pick_core_for_system_placement(placements, chip) 

73 placements.add_placement(Placement(monitor, chip.x, chip.y, p)) 

74 return chip_to_gatherer_map, chip_to_monitor_map
