47def get_estimated_sdram_for_bit_field_region(app_graph, vertex): 

48 """ estimates the SDRAM for the bit field region 

49 

50 :param ~pacman.model.graphs.application.ApplicationGraph app_graph: 

51 the app graph 

52 :param ~pacman.model.graphs.application.ApplicationVertex vertex: 

53 app vertex 

54 :return: the estimated number of bytes used by the bit field region 

55 :rtype: int 

56 """ 

57 sdram = 0 

58 for incoming_edge in app_graph.get_edges_ending_at_vertex(vertex): 

59 if isinstance(incoming_edge, ProjectionApplicationEdge): 

60 edge_pre_vertex = incoming_edge.pre_vertex 

61 max_atoms = determine_max_atoms_for_vertex(edge_pre_vertex) 

62 if incoming_edge.pre_vertex.n_atoms < max_atoms: 

63 max_atoms = incoming_edge.pre_vertex.n_atoms 

64 

65 # Get the number of likely vertices 

66 n_machine_vertices = int(math.ceil( 

67 float(incoming_edge.pre_vertex.n_atoms) / 

68 float(max_atoms))) 

69 n_atoms_per_machine_vertex = int(math.ceil( 

70 float(incoming_edge.pre_vertex.n_atoms) / 

71 n_machine_vertices)) 

72 if isinstance(edge_pre_vertex, AbstractHasDelayStages): 72 ↛ 73line 72 didn't jump to line 73, because the condition on line 72 was never true

73 n_atoms_per_machine_vertex *= \ 

74 edge_pre_vertex.n_delay_stages 

75 n_words_for_atoms = int(math.ceil( 

76 n_atoms_per_machine_vertex / BIT_IN_A_WORD)) 

77 sdram += ( 

78 (ELEMENTS_USED_IN_EACH_BIT_FIELD + ( 

79 n_words_for_atoms * n_machine_vertices)) * 

80 BYTES_PER_WORD) 

81 return sdram

84def get_estimated_sdram_for_key_region(app_graph, vertex): 

85 """ gets an estimate of the bitfield builder region 

86 

87 :param ~pacman.model.graphs.application.ApplicationGraph app_graph: 

88 the app graph 

89 :param ~pacman.model.graphs.application.ApplicationVertex vertex: 

90 app vertex 

91 :return: SDRAM needed 

92 :rtype: int 

93 """ 

94 # basic sdram 

95 sdram = N_KEYS_DATA_SET_IN_WORDS * BYTES_PER_WORD 

96 for in_edge in app_graph.get_edges_ending_at_vertex(vertex): 

97 

98 # Get the number of likely vertices 

99 edge_pre_vertex = in_edge.pre_vertex 

100 max_atoms = determine_max_atoms_for_vertex(edge_pre_vertex) 

101 if in_edge.pre_vertex.n_atoms < max_atoms: 

102 max_atoms = in_edge.pre_vertex.n_atoms 

103 n_edge_vertices = int(math.ceil( 

104 float(in_edge.pre_vertex.n_atoms) / float(max_atoms))) 

105 sdram += (n_edge_vertices * N_ELEMENTS_IN_EACH_KEY_N_ATOM_MAP * 

106 BYTES_PER_WORD) 

107 return sdram

110def _exact_sdram_for_bit_field_region( 

111 machine_graph, vertex, n_key_map): 

112 """ calculates the correct SDRAM for the bitfield region based off \ 

113 the machine graph 

114 

115 :param ~pacman.model.graphs.machine.MachineGraph machine_graph: 

116 machine graph 

117 :param ~pacman.model.graphs.machine.MachineVertex vertex: 

118 the machine vertex 

119 :param ~pacman.model.routing_info.AbstractMachinePartitionNKeysMap \ 

120 n_key_map: 

121 n keys map 

122 :return: SDRAM in bytes 

123 :rtype: int 

124 """ 

125 sdram = ELEMENTS_USED_IN_BIT_FIELD_HEADER * BYTES_PER_WORD 

126 for incoming_edge in machine_graph.get_edges_ending_at_vertex(vertex): 

127 n_keys = n_key_map.n_keys_for_partition( 

128 machine_graph.get_outgoing_partition_for_edge(incoming_edge)) 

129 n_words_for_atoms = int(math.ceil(n_keys / BIT_IN_A_WORD)) 

130 

131 sdram += ( 

132 (ELEMENTS_USED_IN_EACH_BIT_FIELD + n_words_for_atoms) * 

133 BYTES_PER_WORD) 

134 return sdram

161def reserve_bit_field_regions( 

162 spec, machine_graph, n_key_map, vertex, bit_field_builder_region, 

163 bit_filter_region, bit_field_key_region): 

164 """ reserves the regions for the bitfields 

165 

166 :param ~data_specification.DataSpecificationGenerator spec: 

167 dsg spec writer 

168 :param ~pacman.model.graphs.machine.MachineGraph machine_graph: 

169 machine graph 

170 :param ~pacman.model.routing_info.AbstractMachinePartitionNKeysMap \ 

171 n_key_map: 

172 map between partitions and n keys 

173 :param ~pacman.model.graphs.machine.MachineVertex vertex: machine vertex 

174 :param int bit_field_builder_region: region id for the builder region 

175 :param int bit_filter_region: region id for the bitfield region 

176 :param int bit_field_key_region: region id for the key map 

177 """ 

178 

179 # reserve the final destination for the bitfields 

180 spec.reserve_memory_region( 

181 region=bit_filter_region, 

182 size=_exact_sdram_for_bit_field_region( 

183 machine_graph, vertex, n_key_map), 

184 label="bit_field region") 

185 

186 # reserve region for the bitfield builder 

187 spec.reserve_memory_region( 

188 region=bit_field_builder_region, 

189 size=exact_sdram_for_bit_field_builder_region(), 

190 label="bit field builder region") 

191 

192 # reserve memory region for the key region 

193 spec.reserve_memory_region( 

194 region=bit_field_key_region, 

195 size=_exact_sdram_for_bit_field_key_region(machine_graph, vertex), 

196 label="bit field key data")

199def write_bitfield_init_data( 

200 spec, machine_vertex, machine_graph, routing_info, n_key_map, 

201 bit_field_builder_region, master_pop_region_id, 

202 synaptic_matrix_region_id, direct_matrix_region_id, 

203 bit_field_region_id, bit_field_key_map_region_id, 

204 structural_dynamics_region_id, has_structural_dynamics_region): 

205 """ writes the init data needed for the bitfield generator 

206 

207 :param ~data_specification.DataSpecificationGenerator spec: 

208 data spec writer 

209 :param ~pacman.model.graphs.machine.MachineVertex machine_vertex: 

210 machine vertex 

211 :param ~pacman.model.graphs.machine.MachineGraph machine_graph: 

212 machine graph 

213 :param ~pacman.model.routing_info.RoutingInfo routing_info: keys 

214 :param ~pacman.model.routing_info.AbstractMachinePartitionNKeysMap \ 

215 n_key_map: 

216 map for edge to n keys 

217 :param int bit_field_builder_region: the region id for the bitfield builder 

218 :param int master_pop_region_id: the region id for the master pop table 

219 :param int synaptic_matrix_region_id: the region id for the synaptic matrix 

220 :param int direct_matrix_region_id: the region id for the direct matrix 

221 :param int bit_field_region_id: the region id for the bit-fields 

222 :param int bit_field_key_map_region_id: the region id for the key map 

223 :param int structural_dynamics_region_id: the region id for the structural 

224 :param bool has_structural_dynamics_region: 

225 whether the core has a structural_dynamics region 

226 """ 

227 spec.switch_write_focus(bit_field_builder_region) 

228 

229 spec.write_value(master_pop_region_id) 

230 spec.write_value(synaptic_matrix_region_id) 

231 spec.write_value(direct_matrix_region_id) 

232 spec.write_value(bit_field_region_id) 

233 spec.write_value(bit_field_key_map_region_id) 

234 

235 # save 4 bytes by making a key flag of full mask to avoid when not got 

236 # a structural 

237 if not has_structural_dynamics_region: 

238 spec.write_value(FULL_MASK) 

239 else: 

240 spec.write_value(structural_dynamics_region_id) 

241 

242 spec.switch_write_focus(bit_field_key_map_region_id) 

243 

244 # write n keys max atom map 

245 spec.write_value( 

246 len(machine_graph.get_edges_ending_at_vertex(machine_vertex))) 

247 

248 # load in key to max atoms map 

249 for in_coming_edge in machine_graph.get_edges_ending_at_vertex( 

250 machine_vertex): 

251 out_going_partition = \ 

252 machine_graph.get_outgoing_partition_for_edge(in_coming_edge) 

253 spec.write_value( 

254 routing_info.get_first_key_from_partition(out_going_partition)) 

255 spec.write_value( 

256 n_key_map.n_keys_for_partition(out_going_partition)) 

257 

258 # ensure if nothing else that n bitfields in bitfield region set to 0 

259 spec.switch_write_focus(bit_field_region_id) 

260 spec.write_value(0)
