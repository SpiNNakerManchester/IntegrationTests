26class TDMAAwareApplicationVertex(ApplicationVertex): 

27 """ An application vertex that contains the code for using TDMA to spread\ 

28 packet transmission to try to avoid overloading any SpiNNaker routers. 

29 """ 

30 

31 __slots__ = ( 

32 "__initial_offset", 

33 "__n_phases", 

34 "__n_slots", 

35 "__ns_per_cycle", 

36 "__time_between_cores", 

37 "__time_between_spikes") 

38 

39 # 1. initial expected time, 2. min expected time, 3. time between cores 

40 _TDMA_N_ELEMENTS = 3 

41 

42 _TDMA_MISSED_SLOTS_NAME = "Number_of_times_the_tdma_fell_behind" 

43 _TDMA_MISSED_SLOTS_MESSAGE = ( 

44 "The TDMA fell behind by {} times on core {}, {}, {}. " 

45 "try increasing the time_between_cores in the corresponding .cfg")

74 def find_n_phases_for(self, machine_graph, n_keys_map): 

75 """ Compute the number of phases needed for this application vertex. \ 

76 This is the maximum number of packets any machine vertex created \ 

77 by this application vertex can send in one simulation time step. 

78 

79 :param ~pacman.model.graphs.machine.MachineGraph machine_graph: 

80 :param ~pacman.model.routing_info.AbstractMachinePartitionNKeysMap \ 

81 n_keys_map: 

82 :rtype: int 

83 """ 

84 return max( 

85 sum( 

86 n_keys_map.n_keys_for_partition(outgoing_partition) 

87 for outgoing_partition in 

88 machine_graph.get_outgoing_edge_partitions_starting_at_vertex( 

89 machine_vertex)) 

90 for machine_vertex in self.machine_vertices) 

91 

92 def generate_tdma_data_specification_data(self, vertex_index): 

93 """ Generates the TDMA configuration data needed for the data spec 

94 

95 :param int vertex_index: the machine vertex index in the pop 

96 :return: array of data to write. 

97 :rtype: list(int) 

98 """ 

99 core_slot = vertex_index & self.__n_slots 

100 offset_clocks = ( 

101 self.__initial_offset + 

102 (self.__time_between_cores * core_slot * _CLOCKS_PER_NS)) 

103 tdma_clocks = ( 

104 self.__n_phases * self.__time_between_spikes * _CLOCKS_PER_NS) 

105 total_clocks = _CLOCKS_PER_NS * self.__ns_per_cycle 

106 initial_expected_time = total_clocks - offset_clocks 

107 min_expected_time = initial_expected_time - tdma_clocks 

108 clocks_between_sends = self.__time_between_spikes * _CLOCKS_PER_NS 

109 return [initial_expected_time, min_expected_time, 

110 clocks_between_sends] 

111 

112 @property

118 return self._TDMA_N_ELEMENTS * BYTES_PER_WORD
