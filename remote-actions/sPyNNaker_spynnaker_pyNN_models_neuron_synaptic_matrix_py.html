148 def is_direct(self, single_addr): 

149 """ Determine if the given connection can be done with a "direct"\ 

150 synaptic matrix - this must have an exactly 1 entry per row 

151 

152 :param int single_addr: The current offset of the direct matrix 

153 :return: A tuple of a boolean indicating if the matrix is direct and 

154 the next offset of the single matrix 

155 :rtype: (bool, int) 

156 """ 

157 pre_vertex_slice = self.__machine_edge.pre_vertex.vertex_slice 

158 post_vertex_slice = self.__machine_edge.post_vertex.vertex_slice 

159 next_addr = single_addr + self.__single_matrix_size 

160 is_direct = ( 

161 next_addr <= self.__all_single_syn_sz and 

162 not self.is_delayed and 

163 isinstance(self.__synapse_info.connector, OneToOneConnector) and 

164 isinstance(self.__synapse_info.synapse_dynamics, 

165 SynapseDynamicsStatic) and 

166 (pre_vertex_slice.lo_atom == post_vertex_slice.lo_atom) and 

167 (pre_vertex_slice.hi_atom == post_vertex_slice.hi_atom) and 

168 not self.__synapse_info.prepop_is_view and 

169 not self.__synapse_info.postpop_is_view) 

170 return is_direct, next_addr
