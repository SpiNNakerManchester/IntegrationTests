22class SetRouterTimeoutProcess(AbstractMultiConnectionProcess): 

23 """ How to send messages to set router timeouts. These messages need to be\ 

24 sent to cores running the extra monitor binary. 

25 

26 .. note:: 

27 SCAMP sets wait2 to zero by default! 

28 

29 .. note:: 

30 Timeouts are specified in a weird floating point format. 

31 See the SpiNNaker datasheet for details. 

32 """ 

33 

34 def set_wait1_timeout(self, mantissa, exponent, core_subsets): 

35 """ The wait1 timeout is the time from when a packet is received to\ 

36 when emergency routing becomes enabled. 

37 

38 :param int mantissa: Timeout mantissa (0 to 15) 

39 :param int exponent: Timeout exponent (0 to 15) 

40 :param ~spinn_machine.CoreSubsets core_subsets: 

41 Where the extra monitors that manage the routers are located. 

42 """ 

43 for core_subset in core_subsets.core_subsets: 

44 for processor_id in core_subset.processor_ids: 

45 self._set_timeout( 

46 core_subset, processor_id, mantissa, exponent, wait=1) 

47 

48 def set_wait2_timeout(self, mantissa, exponent, core_subsets): 

49 """ The wait2 timeout is the time from when a packet has emergency\ 

50 routing enabled for it to when it is dropped. 

51 

52 :param int mantissa: Timeout mantissa (0 to 15) 

53 :param int exponent: Timeout exponent (0 to 15) 

54 :param ~spinn_machine.CoreSubsets core_subsets: 

55 Where the extra monitors that manage the routers are located. 

56 """ 

57 for core_subset in core_subsets.core_subsets: 

58 for processor_id in core_subset.processor_ids: 

59 self._set_timeout( 

60 core_subset, processor_id, mantissa, exponent, wait=2) 

61 

62 def _set_timeout(self, core, processor_id, mantissa, exponent, wait): 

63 """ Set a timeout for a router controlled by an extra monitor on a\ 

64 core. This is not a parallelised operation in order to aid\ 

65 debugging when it fails. 

66 

67 :param ~spinn_machine.CoreSubset core: 

68 :param int processor_id: 

69 :param int mantissa: 

70 :param int exponent: 

71 :param int wait: 

72 """ 

73 self._send_request(SetRouterTimeoutMessage( 

74 core.x, core.y, processor_id, 

75 timeout_mantissa=mantissa, timeout_exponent=exponent, wait=wait))
