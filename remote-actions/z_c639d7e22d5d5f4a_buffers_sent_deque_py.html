15from collections import deque 

16from threading import Lock 

17from typing import Deque, Iterable 

18import logging 

19from spinn_utilities.log import FormatAdapter 

20from spinnman.messages.eieio.command_messages import ( 

21 EventStopRequest, HostSendSequencedData) 

22from spinnman.messages.eieio import AbstractEIEIOMessage 

23from spinn_front_end_common.utilities.exceptions import SpinnFrontEndException 

24 

25logger = FormatAdapter(logging.getLogger(__name__)) 

26 

27#: The total number of sequence numbers 

28_N_SEQUENCES = 256 

29 

30 

31class BuffersSentDeque(object): 

32 """ 

33 A tracker of buffers sent / to send for a region 

34 """ 

35 

36 __slots__ = ( 

37 #: The region being managed 

38 "_region", 

39 

40 #: A queue of messages sent, ordered by sequence number 

41 "_buffers_sent", 

42 

43 #: The current sequence number of the region 

44 "_sequence_number", 

45 

46 #: A lock for the sequence number 

47 "_sequence_lock", 

48 

49 #: The last sequence number to be received on the machine 

50 "_last_received_sequence_number", 

51 

52 #: True if the stop message has been sent 

53 "_sent_stop_message", 

54 

55 #: The number of sequence numbers allowed in a single transmission 

56 "_n_sequences_per_transmission") 

57 

58 def __init__(self, region: int, sent_stop_message: bool = False, 

59 n_sequences_per_tranmission: int = 64): 

60 """ 

61 :param region: The region being managed 

62 :param sent_stop_message: True if the stop message has been sent 

63 :param n_sequences_per_tranmission: 

64 The number of sequences allowed in each transmission set 

65 """ 

66 self._region = region 

67 self._buffers_sent: Deque[HostSendSequencedData] = deque( 

68 maxlen=n_sequences_per_tranmission) 

69 self._sequence_number = 0 

70 self._sequence_lock = Lock() 

71 self._last_received_sequence_number = _N_SEQUENCES - 1 

72 self._sent_stop_message = sent_stop_message 

73 self._n_sequences_per_transmission = n_sequences_per_tranmission 

74 

75 @property 

76 def is_full(self) -> bool: 

77 """ 

78 Whether the number of messages sent is at the limit for the 

79 sequencing system. 

80 """ 

81 return len(self._buffers_sent) >= self._n_sequences_per_transmission 

82 

83 def is_empty(self) -> bool: 

84 """ 

85 Determine if there are no messages. 

86 """ 

87 return len(self._buffers_sent) == 0 

88 

89 def send_stop_message(self) -> None: 

90 """ 

91 Send a message to indicate the end of all the messages. 

92 """ 

93 if not self._sent_stop_message: 

94 self._sent_stop_message = True 

95 self.add_message_to_send(EventStopRequest()) 

96 

97 def add_message_to_send(self, message: AbstractEIEIOMessage) -> None: 

98 """ 

99 Add a message to send. The message is converted to a sequenced 

100 message. 

101 

102 :param message: The message to be added 

103 """ 

104 # If full, raise an exception 

105 if self.is_full: 

106 raise SpinnFrontEndException("The buffer is full") 

107 

108 # Create a sequenced message and update the sequence number 

109 self._sequence_lock.acquire() 

110 sequenced_message = HostSendSequencedData( 

111 self._region, self._sequence_number, message) 

112 self._sequence_number = (self._sequence_number + 1) % _N_SEQUENCES 

113 self._sequence_lock.release() 

114 

115 # Add the sequenced message to the buffers 

116 self._buffers_sent.append(sequenced_message) 

117 

118 @property 

119 def messages(self) -> Iterable[HostSendSequencedData]: 

120 """ 

121 The messages that have been added to the set. 

122 """ 

123 return self._buffers_sent 

124 

125 def update_last_received_sequence_number( 

126 self, last_received_sequence_no: int) -> bool: 

127 """ 

128 Updates the last received sequence number. If the sequence number is 

129 within the valid window, packets before the sequence number within the 

130 window are removed, and the last received sequence number is updated, 

131 thus moving the window for the next call. If the sequence number is 

132 not within the valid window, it is assumed to be invalid and so is 

133 ignored. 

134 

135 :param last_received_sequence_no: The new sequence number 

136 :return: True if update went ahead, False if it was ignored 

137 """ 

138 # The sequence number window is between the last received and 

139 # the last received + window size, taking account that the end 

140 # of the window might wrap 

141 min_seq_no_acceptable = self._last_received_sequence_number 

142 max_seq_no_acceptable = ( 

143 (min_seq_no_acceptable + self._n_sequences_per_transmission) 

144 % _N_SEQUENCES) 

145 

146 if (min_seq_no_acceptable <= last_received_sequence_no <= 

147 max_seq_no_acceptable): 

148 # The sequence hasn't wrapped and the sequence is valid 

149 self._last_received_sequence_number = last_received_sequence_no 

150 self._remove_messages() 

151 return True 

152 elif max_seq_no_acceptable < min_seq_no_acceptable: 

153 # The sequence has wrapped 

154 if (0 <= last_received_sequence_no <= max_seq_no_acceptable or 

155 min_seq_no_acceptable <= last_received_sequence_no <= 

156 _N_SEQUENCES): 

157 # The sequence is in the valid range 

158 self._last_received_sequence_number = last_received_sequence_no 

159 self._remove_messages() 

160 return True 

161 

162 # If none of the above match, the sequence is out of the window 

163 return False 

164 

165 def _remove_messages(self) -> None: 

166 """ 

167 Remove messages that are no longer relevant, based on the last 

168 sequence number received. 

169 """ 

170 min_sequence = (self._last_received_sequence_number - 

171 self._n_sequences_per_transmission) 

172 logger.debug("Removing buffers between {} and {}", 

173 min_sequence, self._last_received_sequence_number) 

174 

175 # If we are at the start of the sequence numbers, keep going back up to 

176 # the allowed window 

177 if min_sequence < 0: 

178 back_min_sequence = min_sequence + _N_SEQUENCES 

179 while (self._buffers_sent and 

180 self._buffers_sent[0].sequence_no > back_min_sequence): 

181 logger.debug("Removing buffer with sequence {}", 

182 self._buffers_sent[0].sequence_no) 

183 self._buffers_sent.popleft() 

184 

185 # Go back through the queue until we reach the last received sequence 

186 while (self._buffers_sent and 

187 min_sequence < self._buffers_sent[0].sequence_no <= 

188 self._last_received_sequence_number): 

189 logger.debug("Removing buffer with sequence {}", 

190 self._buffers_sent[0].sequence_no) 

191 self._buffers_sent.popleft()
