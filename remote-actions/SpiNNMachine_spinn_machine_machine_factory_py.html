91def machine_from_chips(chips): 

92 """ 

93 Create a machine with the assumed wrap-around based on the sizes. 

94 

95 The size of the machine is calculated from the list of chips. 

96 

97 :param chips: Full list of all chips on this machine. 

98 Or at least a list which includes a chip with the highest X and

100 :type chips: list(Chip) 

101 :return: A subclass of Machine 

102 :rtype: Machine 

103 """ 

104 max_x = 0 

105 max_y = 0

111 return machine_from_size(max_x + 1, max_y + 1, chips)

114def _machine_ignore(original, dead_chips, dead_links): 

115 """ Creates a near copy of the machine without the dead bits. 

116 

117 Creates a new Machine with the the Chips that where in the original 

118 machine but are not listed as dead. 

119 

120 Each Chip will only have the links that already existed and are not listed 

121 as dead. 

122 

123 Spinnaker_links and fpga_links are re-added so removing a wrap around link 

124 could results in and extra spinnaker or fpga link. 

125 

126 Dead Chips or links not in the original machine are ignored. 

127 

128 Does not change the original machine! 

129 

130 :param original: Machine to make a near copy of 

131 :param dead_chips: Collection of dead chips x and y cooridnates 

132 :type dead_chips: Collection (int, int) 

133 :param dead_links: Collection of dead link x y and direction cooridnates 

134 :type dead_links: Collection of (int, int, int) 

135 :return: A New Machine object 

136 """ 

137 new_machine = machine_from_size(original.width, original.height) 

138 links_map = defaultdict(set) 

139 for x, y, d, _ in dead_links: 

140 links_map[(x, y)].add(d) 

141 for chip in original.chips: 

142 if (chip.x, chip.y) in dead_chips: 

143 continue 

144 if (chip.x, chip.y) in links_map: 

145 links = [] 

146 for link in chip.router.links: 

147 if link.source_link_id not in links_map[(chip.x, chip.y)]: 

148 links.append(link) 

149 router = Router(links, chip.router.emergency_routing_enabled, 

150 chip.router.n_available_multicast_entries) 

151 chip = Chip( 

152 chip.x, chip.y, chip.n_processors, router, chip.sdram, 

153 chip.nearest_ethernet_x, chip.nearest_ethernet_y, 

154 chip.ip_address, chip.virtual, chip.tag_ids) 

155 new_machine.add_chip(chip) 

156 new_machine.add_spinnaker_links() 

157 new_machine.add_fpga_links() 

158 new_machine.validate() 

159 return new_machine
