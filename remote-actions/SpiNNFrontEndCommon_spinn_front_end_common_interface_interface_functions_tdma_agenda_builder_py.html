23class TDMAAgendaBuilder(object): 

24 """ Algorithm that builds an agenda for transmissions. It uses a TDMA\ 

25 (time-division multiple access) system and graph colouring to deduce\ 

26 the agenda set up. Ensures parallel transmissions so that the\ 

27 destination should never be overloaded. 

28 

29 :param ~pacman.model.graphs.machine.MachineGraph machine_graph: 

30 the graph for which we are planning an agenda 

31 :param int number_of_cpu_cycles_per_receive: 

32 how long the packet reception callback takes in CPU cycles 

33 :param int other_cpu_demands_in_cpu_cycles: 

34 extra costs (e.g. timer tick callback etc.) 

35 :param int n_packets_per_time_window: 

36 how many packets are to be sent per time window 

37 :param int machine_time_step: the timer tick in microseconds 

38 :param int time_scale_factor: 

39 the multiplicative factor on the machine time step. 

40 :param float safety_factor: the end user safety factor 

41 :return: 

42 the agenda for each vertex on its time window and its offset 

43 between spike transmissions 

44 :rtype: dict(~pacman.model.graphs.machine.MachineVertex,dict(str,int)) 

45 :raises ConfigurationException: If colouring fails 

46 """ 

47 

48 def __call__( 

49 self, machine_graph, number_of_cpu_cycles_per_receive, 

50 other_cpu_demands_in_cpu_cycles, n_packets_per_time_window, 

51 machine_time_step, time_scale_factor, safety_factor=1): 

52 """ 

53 :param ~.MachineGraph machine_graph: 

54 :param int number_of_cpu_cycles_per_receive: 

55 :param int other_cpu_demands_in_cpu_cycles: 

56 :param int n_packets_per_time_window: 

57 :param int machine_time_step: 

58 :param int time_scale_factor: 

59 :param float safety_factor: 

60 :rtype: dict(~.MachineVertex,dict(str,int)) 

61 """ 

62 # pylint: disable=too-many-arguments 

63 

64 # figure out max in edges (as this becomes the colour scheme 

65 max_in_edges = self._calculate_max_edges(machine_graph) 

66 

67 # check its possible to make an agenda 

68 cycles_per_window = self._check_time_window_size( 

69 number_of_cpu_cycles_per_receive, machine_time_step, 

70 time_scale_factor, n_packets_per_time_window, max_in_edges, 

71 safety_factor, other_cpu_demands_in_cpu_cycles) 

72 

73 # do graph colouring to ensure TDMA is met 

74 time_offset = self._handle_colouring(max_in_edges, machine_graph) 

75 

76 # build the agenda 

77 return self._build_agenda( 

78 machine_graph, cycles_per_window, time_offset, 

79 n_packets_per_time_window)

81 @staticmethod 

82 def _build_agenda( 

83 machine_graph, cycles_per_window, time_offset, packets_per_window): 

84 """ Builds the TDMA system. 

85 

86 :param ~.MachineGraph machine_graph: 

87 the machine graph of the application 

88 :param int cycles_per_window: 

89 how long the receive function takes 

90 :param int time_offset: the colouring offsets 

91 :param int packets_per_window: 

92 the packets expected to be sent per window 

93 :return: 

94 the agenda for each vertex on its time window and its offset 

95 between spike transmissions 

96 :rtype: dict(~.MachineVertex,dict(str,int)) 

97 """ 

98 # pylint: disable=too-many-arguments 

99 return { 

100 vertex: { 

101 'time_offset': cycles_per_window * time_offset[vertex], 

102 'time_between_packets': cycles_per_window // packets_per_window 

103 } 

104 for vertex in machine_graph.vertices}

106 @staticmethod 

107 def _calculate_max_edges(machine_graph): 

108 """ Deduces the max incoming edges for any vertex 

109 

110 :param ~.MachineGraph machine_graph: 

111 the machine graph of the application 

112 :return: 

113 the max number of incoming edges to any vertex in the application 

114 :rtype: int 

115 """ 

116 if not machine_graph.vertices: 

117 # Python 2.7 support 

118 return 0 

119 return max( 

120 len(machine_graph.get_edges_ending_at_vertex(vertex)) 

121 for vertex in machine_graph.vertices)

123 def _handle_colouring(self, max_in_edges, machine_graph): 

124 """ Operates the graph colouring greedy code. 

125 

126 :param int max_in_edges: the number of colours to colour the graph 

127 :param ~.MachineGraph machine_graph: 

128 the machine graph representation of the application 

129 :return: vertex to time offset mapping 

130 :rtype: dict(~.MachineVertex, int) 

131 """ 

132 time_offset = dict() 

133 colours = list(range(0, max_in_edges + 1)) # plus 1 to include myself 

134 # use greedy colouring algorithm to colour graph 

135 for vertex in machine_graph.vertices: 

136 self._colour_vertex(vertex, machine_graph, colours, time_offset) 

137 return time_offset

139 def _colour_vertex(self, vertex, machine_graph, colours, colour_mapping): 

140 """ Cycles though available colours to find a colour that can be used\ 

141 for a particular vertex, and assigns it once it finds one. 

142 

143 :param ~.MachineVertex vertex: the vertex to be coloured 

144 :param ~.MachineGraph machine_graph: 

145 the machine graph representation of the application 

146 :param list(int) colours: 

147 the available colours for the graph colouring 

148 :param dict(~.MachineVertex,int) colour_mapping: 

149 the mapping between vertex and colour 

150 :raises ConfigurationException: If no colouring can be found 

151 """ 

152 for colour in colours: 

153 if not any( 

154 colour_mapping.get(edge.pre_vertex, -1) == colour 

155 for edge in machine_graph.get_edges_ending_at_vertex( 

156 vertex)): 

157 colour_mapping[vertex] = colour 

158 return 

159 raise ConfigurationException( 

160 "cannot colour this edge. Something screwed up")

162 @staticmethod 

163 def _check_time_window_size( 

164 number_of_cpu_cycles_per_receive, machine_time_step, 

165 time_scale_factor, n_packets_per_time_window, max_in_edges, 

166 safety_factor, other_cpu_demands_in_cpu_cycles): 

167 """ Calculates the CPU cycles per window, and therefore verifies if\ 

168 there is enough time to do so with a end user safety margin 

169 

170 :param int number_of_cpu_cycles_per_receive: 

171 how long the packet reception callback takes in CPU cycles 

172 :param int machine_time_step: the timer tick in microseconds 

173 :param int time_scale_factor: 

174 the multiplicative factor on the machine time step. 

175 :param float n_packets_per_time_window: 

176 how many packets are to be sent per time window 

177 :param int max_in_edges: 

178 the max number of edges going into any vertex in the machine graph 

179 :param float safety_factor: the end user safely factor 

180 :param int other_cpu_demands_in_cpu_cycles: 

181 extra costs (e.g. timer tick callback etc.) 

182 :return: CPU cycles available per window. 

183 :rtype: float 

184 :raises ConfigurationException: 

185 if the overall time is below what is possible to receive packets 

186 with 

187 """ 

188 # pylint: disable=too-many-arguments 

189 

190 # figure out if its feasible for window to work 

191 total_cycles_available = ( 

192 machine_time_step * time_scale_factor * 

193 _CONVERSION_BETWEEN_MICRO_TO_CPU_CYCLES) 

194 

195 cpu_cycles_needed_per_window = math.ceil( 

196 (number_of_cpu_cycles_per_receive + 

197 _CPU_CYCLES_PER_INTERRUPT_TO_CALLBACK) * 

198 (n_packets_per_time_window + safety_factor)) 

199 

200 time_needed_per_epoch = ( 

201 (cpu_cycles_needed_per_window * max_in_edges) + 

202 other_cpu_demands_in_cpu_cycles) 

203 

204 if total_cycles_available < time_needed_per_epoch: 

205 raise ConfigurationException( 

206 "Cannot create a window for this simulation with its " 

207 "combined machine_time_step, time_scale_factor, and time " 

208 "needed per packet receive and the number of edges going to " 

209 "a core. Recommend reducing the graph connectivity, " 

210 "or increasing the machine time step or the time scale " 

211 "factor.\n\n" 

212 "Assuming same connectivity, the best current available " 

213 "would be {} machine_time_steps with a timescale factor of " 

214 "1".format(time_needed_per_epoch)) 

215 return cpu_cycles_needed_per_window
