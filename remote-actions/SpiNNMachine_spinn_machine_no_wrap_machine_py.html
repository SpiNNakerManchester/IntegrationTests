20class NoWrapMachine(Machine): 

21 # pylint: disable=useless-super-delegation 

22 def __init__(self, width, height, chips=None, origin=None): 

23 """ Creates an machine without wrap-arounds. 

24 

25 :param width: The width of the machine excluding any virtual chips 

26 :param height: The height of the machine excluding any virtual chips 

27 :param chips: An iterable of chips in the machine 

28 :type chips: iterable of :py:class:`~spinn_machine.Chip` 

29 :param origin: Extra information about how this mnachine was created \ 

30 to be used in the str method. Example "Virtual" or "Json" 

31 :raise spinn_machine.exceptions.SpinnMachineAlreadyExistsException: \ 

32 If any two chips have the same x and y coordinates 

33 """ 

34 super(NoWrapMachine, self).__init__(width, height, chips, origin)

36 @overrides(Machine.multiple_48_chip_boards) 

37 def multiple_48_chip_boards(self): 

38 return (self._width - 4) % 12 == 0 and (self._height - 4) % 12 == 0

40 @overrides(Machine.get_xys_by_ethernet) 

41 def get_xys_by_ethernet(self, ethernet_x, ethernet_y): 

42 for (x, y) in self._local_xys: 

43 yield (x + ethernet_x, y + ethernet_y) 

44 

45 @overrides(Machine.get_xy_cores_by_ethernet) 

46 def get_xy_cores_by_ethernet(self, ethernet_x, ethernet_y): 

47 if (self._width == self._height == 8) or \ 

48 self.multiple_48_chip_boards(): 

49 for (x, y), n_cores in self.CHIPS_PER_BOARD.items(): 

50 # if ethernet_x/y != 0 GIGO mode so ignore ethernet 

51 yield ((x + ethernet_x, y + ethernet_y), n_cores) 

52 else: 

53 # covers weird sizes 

54 n_cores = Machine.max_cores_per_chip() 

55 for (x, y) in self._local_xys: 

56 yield((x, y), n_cores)

58 @overrides(Machine.get_chips_by_ethernet) 

59 def get_chips_by_ethernet(self, ethernet_x, ethernet_y): 

60 for (x, y) in self._local_xys: 

61 chip_xy = (x + ethernet_x, y + ethernet_y) 

62 if (chip_xy) in self._chips: 

63 yield self._chips[chip_xy]

67 for (x, y) in self._local_xys:

72 @overrides(Machine.get_down_xys_by_ethernet) 

73 def get_down_xys_by_ethernet(self, ethernet_x, ethernet_y): 

74 for (x, y) in self._local_xys: 

75 chip_xy = ((x + ethernet_x),

80 @overrides(Machine.xy_over_link) 

81 def xy_over_link(self, x, y, link): 

82 add_x, add_y = Machine.LINK_ADD_TABLE[link] 

83 link_x = x + add_x

88 def get_local_xy(self, chip): 

89 local_x = chip.x - chip.nearest_ethernet_x 

90 local_y = chip.y - chip.nearest_ethernet_y

93 @overrides(Machine.get_global_xy) 

94 def get_global_xy(self, local_x, local_y, ethernet_x, ethernet_y):

99 @overrides(Machine.get_vector_length) 

100 def get_vector_length(self, source, destination): 

101 x = destination[0] - source[0] 

102 y = destination[1] - source[1] 

103 

104 # When vectors are minimised, (1,1,1) is added or subtracted from them. 

105 # This process does not change the range of numbers in the vector. 

106 # When a vector is minimal, 

107 # it is easy to see that the range of numbers gives the 

108 # magnitude since there are at most two non-zero numbers (with opposite 

109 # signs) and the sum of their magnitudes will also be their range. 

110 # 

111 # Though ideally this code would be written:: 

112 # 

113 # >>> return max(x, y, z) - min(x, y, z) 

114 

115 # This can be farther optimised with then knowledge that z is always 0 

116 # An x and y having the samne sign they can be replaced with a z 

117 # IE: Replace a North and an East with a NorthEast 

118 # So the length is the greater absolutule value of x or y 

119 # If the are opossite use the sum of the absolute values 

120 

121 if x > 0: 

122 if y > 0: 

123 # the greater abs 

124 if x > y: 

125 return x 

126 else: 

127 return y 

128 else: 

129 # abs(positve x) + abs(negative y) 

130 return x - y 

131 else: 

132 if y > 0: 

133 return y - x 

134 else: 

135 # the greater abs 

136 if x > y: 

137 return - y 

138 else: 

139 return - x
