226 @overrides(ProvidesProvenanceDataFromMachineImpl. 

227 get_provenance_data_from_machine) 

228 def get_provenance_data_from_machine(self, transceiver, placement): 

229 provenance_data = self._read_provenance_data(transceiver, placement) 

230 provenance_items = self._read_basic_provenance_items( 

231 provenance_data, placement) 

232 provenance_data = self._get_remaining_provenance_data_items( 

233 provenance_data) 

234 

235 times_timer_tic_overran = 0 

236 for item in provenance_items: 

237 if item.names[-1] == self._TIMER_TICK_OVERRUN: 

238 times_timer_tic_overran = item.value 

239 

240 n_saturations = provenance_data[ 

241 self.EXTRA_PROVENANCE_DATA_ENTRIES.SATURATION_COUNT.value] 

242 n_buffer_overflows = provenance_data[ 

243 self.EXTRA_PROVENANCE_DATA_ENTRIES.BUFFER_OVERFLOW_COUNT.value] 

244 n_pre_synaptic_events = provenance_data[ 

245 self.EXTRA_PROVENANCE_DATA_ENTRIES.PRE_SYNAPTIC_EVENT_COUNT.value] 

246 last_timer_tick = provenance_data[ 

247 self.EXTRA_PROVENANCE_DATA_ENTRIES.CURRENT_TIMER_TIC.value] 

248 n_plastic_saturations = provenance_data[ 

249 self.EXTRA_PROVENANCE_DATA_ENTRIES. 

250 PLASTIC_SYNAPTIC_WEIGHT_SATURATION_COUNT.value] 

251 n_ghost_searches = provenance_data[ 

252 self.EXTRA_PROVENANCE_DATA_ENTRIES.GHOST_POP_TABLE_SEARCHES.value] 

253 failed_to_read_bit_fields = provenance_data[ 

254 self.EXTRA_PROVENANCE_DATA_ENTRIES.FAILED_TO_READ_BIT_FIELDS.value] 

255 dma_completes = provenance_data[ 

256 self.EXTRA_PROVENANCE_DATA_ENTRIES.DMA_COMPLETES.value] 

257 spike_processing_count = provenance_data[ 

258 self.EXTRA_PROVENANCE_DATA_ENTRIES.SPIKE_PROGRESSING_COUNT.value] 

259 invalid_master_pop_hits = provenance_data[ 

260 self.EXTRA_PROVENANCE_DATA_ENTRIES.INVALID_MASTER_POP_HITS.value] 

261 n_packets_filtered_by_bit_field_filter = provenance_data[ 

262 self.EXTRA_PROVENANCE_DATA_ENTRIES.BIT_FIELD_FILTERED_COUNT.value] 

263 n_rewires = provenance_data[ 

264 self.EXTRA_PROVENANCE_DATA_ENTRIES.N_REWIRES.value] 

265 n_late_packets = provenance_data[ 

266 self.EXTRA_PROVENANCE_DATA_ENTRIES.N_LATE_SPIKES.value] 

267 input_buffer_max_filled_size = provenance_data[ 

268 self.EXTRA_PROVENANCE_DATA_ENTRIES.INPUT_BUFFER_FILLED_SIZE.value] 

269 tdma_misses = provenance_data[ 

270 self.EXTRA_PROVENANCE_DATA_ENTRIES.TDMA_MISSES.value] 

271 

272 label, x, y, p, names = self._get_placement_details(placement) 

273 

274 # translate into provenance data items 

275 provenance_items.append(ProvenanceDataItem( 

276 self._add_name(names, self.SATURATION_COUNT_NAME), 

277 n_saturations, report=n_saturations > 0, 

278 message=self.SATURATION_COUNT_MESSAGE.format( 

279 label, x, y, p, n_saturations))) 

280 provenance_items.append(ProvenanceDataItem( 

281 self._add_name(names, self.INPUT_BUFFER_FULL_NAME), 

282 n_buffer_overflows, report=n_buffer_overflows > 0, 

283 message=self.INPUT_BUFFER_FULL_MESSAGE.format( 

284 label, x, y, p, n_buffer_overflows))) 

285 provenance_items.append(ProvenanceDataItem( 

286 self._add_name(names, self.TOTAL_PRE_SYNAPTIC_EVENT_NAME), 

287 n_pre_synaptic_events)) 

288 provenance_items.append(ProvenanceDataItem( 

289 self._add_name(names, self.LAST_TIMER_TICK_NAME), 

290 last_timer_tick)) 

291 provenance_items.append(ProvenanceDataItem( 

292 self._add_name(names, self.SATURATED_PLASTIC_WEIGHTS_NAME), 

293 n_plastic_saturations, report=n_plastic_saturations > 0, 

294 message=self.SATURATED_PLASTIC_WEIGHTS_MESSAGE.format( 

295 label, x, y, p, n_plastic_saturations))) 

296 provenance_items.append(ProvenanceDataItem( 

297 self._add_name(names, self.N_RE_WIRES_NAME), n_rewires)) 

298 provenance_items.append(ProvenanceDataItem( 

299 self._add_name(names, self.GHOST_SEARCHES), n_ghost_searches, 

300 report=n_ghost_searches > 0, 

301 message=( 

302 "The number of failed population table searches for {} on {}," 

303 " {}, {} was {}. If this number is large relative to the " 

304 "predicted incoming spike rate, try increasing source and " 

305 "target neurons per core".format( 

306 label, x, y, p, n_ghost_searches)))) 

307 provenance_items.append(ProvenanceDataItem( 

308 self._add_name(names, self.BIT_FIELDS_NOT_READ), 

309 failed_to_read_bit_fields, report=False, 

310 message=( 

311 "The filter for stopping redundant DMA's couldn't be fully " 

312 "filled in, it failed to read {} entries, which means it " 

313 "required a max of {} extra bytes of DTCM (assuming cores " 

314 "have at max 255 neurons. Try reducing neurons per core, or " 

315 "size of buffers, or neuron params per neuron etc.".format( 

316 failed_to_read_bit_fields, 

317 failed_to_read_bit_fields * 

318 self.WORDS_TO_COVER_256_ATOMS)))) 

319 provenance_items.append(ProvenanceDataItem( 

320 self._add_name(names, self.DMA_COMPLETE), dma_completes)) 

321 provenance_items.append(ProvenanceDataItem( 

322 self._add_name(names, self.SPIKES_PROCESSED), 

323 spike_processing_count)) 

324 provenance_items.append(ProvenanceDataItem( 

325 self._add_name(names, self.INVALID_MASTER_POP_HITS), 

326 invalid_master_pop_hits, report=invalid_master_pop_hits > 0, 

327 message=( 

328 "There were {} keys which were received by core {}:{}:{} which" 

329 " had no master pop entry for it. This is a error, which most " 

330 "likely strives from bad routing.".format( 

331 invalid_master_pop_hits, x, y, p)))) 

332 provenance_items.append((ProvenanceDataItem( 

333 self._add_name(names, self.BIT_FIELD_FILTERED_PACKETS), 

334 n_packets_filtered_by_bit_field_filter, 

335 report=( 

336 n_packets_filtered_by_bit_field_filter > 0 and ( 

337 n_buffer_overflows > 0 or 

338 times_timer_tic_overran > 0)), 

339 message=( 

340 "There were {} packets received by {}:{}:{} that were " 

341 "filtered by the Bitfield filterer on the core. These packets " 

342 "were having to be stored and processed on core, which means " 

343 "the core may not be running as efficiently as it could. " 

344 "Please adjust the network or the mapping so that these " 

345 "packets are filtered in the router to improve " 

346 "performance.".format( 

347 n_packets_filtered_by_bit_field_filter, x, y, p))))) 

348 late_message = ( 

349 self._N_LATE_SPIKES_MESSAGE_DROP if self.__drop_late_spikes 

350 else self._N_LATE_SPIKES_MESSAGE_NO_DROP) 

351 provenance_items.append(ProvenanceDataItem( 

352 self._add_name(names, self._N_LATE_SPIKES_NAME), 

353 n_late_packets, report=n_late_packets > 0, 

354 message=late_message.format(n_late_packets, label, x, y, p))) 

355 

356 provenance_items.append(ProvenanceDataItem( 

357 self._add_name(names, self._MAX_FILLED_SIZE_OF_INPUT_BUFFER_NAME), 

358 input_buffer_max_filled_size, report=False)) 

359 

360 provenance_items.append(self._app_vertex.get_tdma_provenance_item( 

361 names, x, y, p, tdma_misses)) 

362 return provenance_items
