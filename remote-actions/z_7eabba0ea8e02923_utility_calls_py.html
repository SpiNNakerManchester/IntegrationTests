22def expand_to_bit_array(value: int) -> numpy.ndarray: 

23 """ 

24 Expand a 32-bit value in to an array of length 32 of uint8 values, 

25 each of which is a 1 or 0. 

26 

27 :param value: The value to expand 

28 """ 

29 return numpy.unpackbits( 

30 numpy.asarray([value], dtype=">u4").view(dtype="uint8"))

33def compress_from_bit_array(bit_array: numpy.ndarray) -> int: 

34 """ 

35 Compress a bit array of 32 uint8 values, where each is a 1 or 0, 

36 into a 32-bit value. 

37 

38 :param bit_array: The array to compress 

39 """ 

40 return numpy.packbits(bit_array).view(dtype=">u4")[0].item()

43def compress_bits_from_bit_array( 

44 bit_array: numpy.ndarray, bit_positions: numpy.ndarray) -> int: 

45 """ 

46 Compress specific positions from a bit array of 32 uint8 value, 

47 where is a 1 or 0, into a 32-bit value. 

48 

49 :param bit_array: 

50 The array to extract the value from 

51 :param bit_positions: 

52 The positions of the bits to extract, 

53 each value being between 0 and 31 

54 """ 

55 expanded_value = numpy.zeros(32, dtype="uint8") 

56 expanded_value[-len(bit_positions):] = bit_array[bit_positions] 

57 return compress_from_bit_array(expanded_value)

68def is_single(iterable: Iterable[Any]) -> bool: 

69 """ 

70 Test if there is exactly one item in the iterable. 

71 """ 

72 iterator = iter(iterable) 

73 

74 # Test if there is a first item, if not return False 

75 if next(iterator, None) is None: 

76 return False 

77 

78 # Test if there is a second item, if not return True 

79 if next(iterator, None) is None: 

80 return True 

81 

82 # Otherwise return False 

83 return False

95def get_key_ranges(key: int, mask: int) -> Iterable[Tuple[int, int]]: 

96 """ 

97 Get a generator of base_key, n_keys pairs that represent ranges 

98 allowed by the mask. 

99 

100 :param key: The base key 

101 :param mask: The mask 

102 """ 

103 unwrapped_mask = expand_to_bit_array(mask) 

104 first_zeros = list() 

105 remaining_zeros = list() 

106 pos = len(unwrapped_mask) - 1 

107 

108 # Keep the indices of the first set of zeros 

109 while pos >= 0 and unwrapped_mask[pos] == 0: 

110 first_zeros.append(pos) 

111 pos -= 1 

112 

113 # Find all the remaining zeros 

114 while pos >= 0: 

115 if unwrapped_mask[pos] == 0: 

116 remaining_zeros.append(pos) 

117 pos -= 1 

118 

119 # Loop over 2^len(remaining_zeros) to produce the base key, 

120 # with n_keys being 2^len(first_zeros) 

121 n_sets = 2 ** len(remaining_zeros) 

122 n_keys = 2 ** len(first_zeros) 

123 if not remaining_zeros: 

124 yield key, n_keys 

125 return 

126 unwrapped_key = expand_to_bit_array(key) 

127 for value in range(n_sets): 

128 generated_key = numpy.copy(unwrapped_key) 

129 generated_key[remaining_zeros] = \ 

130 expand_to_bit_array(value)[-len(remaining_zeros):] 

131 yield compress_from_bit_array(generated_key), n_keys

134def get_n_bits(n_values: int) -> int: 

135 """ 

136 Determine how many bits are required for the given number of values. 

137 

138 :param n_values: the number of values (starting at 0) 

139 :return: the number of bits required to express that many values 

140 """ 

141 if n_values == 0: 

142 return 0 

143 if n_values == 1: 

144 return 1 

145 return int(math.ceil(math.log2(n_values)))

160def get_keys( 

161 base_key: int, vertex_slice: Slice, 

162 n_extra_bits: int = 0) -> numpy.ndarray: 

163 """ 

164 Get the keys for a given vertex slice. 

165 

166 :param base_key: The base key for the vertex slice 

167 :param vertex_slice: The slice of the vertex to get keys for 

168 :param n_extra_bits: Additional right shift to apply to atoms 

169 """ 

170 indices = numpy.arange(0, vertex_slice.n_atoms) << n_extra_bits 

171 return base_key + indices
