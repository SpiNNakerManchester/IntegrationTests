30class CSAConnector(AbstractConnector): 

31 """ Make connections using a Connection Set Algebra (Djurfeldt 2012)\ 

32 description between the neurons in the pre- and post-populations. 

33 If you get TypeError in Python 3 see: 

34 https://github.com/INCF/csa/issues/10 

35 """ 

36 

37 __slots = [ 

38 "__cset", "__full_connection_set", "__full_cset"] 

39 

40 def __init__(self, cset, safe=True, callback=None, verbose=False): 

41 """ 

42 :param csa.connset.CSet cset: 

43 A description of the connection set between populations 

44 :param bool safe: 

45 :param callable callback: Ignored 

46 :param bool verbose: 

47 :raises ImportError: 

48 if the `csa` library isn't present; it's tricky to install in 

49 some environments so we don't force it to be present unless you 

50 want to actually use this class. 

51 """ 

52 super(CSAConnector, self).__init__(safe, callback, verbose) 

53 found, ex = _csa_found 

54 if not found: 

55 raise ex 

56 self.__cset = cset 

57 

58 # Storage for full connection sets 

59 self.__full_connection_set = None 

60 self.__full_cset = None 

61 

62 @overrides(AbstractConnector.get_delay_maximum) 

63 def get_delay_maximum(self, synapse_info): 

64 n_conns_max = synapse_info.n_pre_neurons * synapse_info.n_post_neurons 

65 # we can probably look at the array and do better than this? 

66 return self._get_delay_maximum(synapse_info.delays, n_conns_max) 

67 

68 def _get_n_connections( 

69 self, pre_vertex_slice, post_vertex_slice, synapse_info): 

70 """ 

71 :param ~pacman.model.graphs.common.Slice pre_vertex_slice: 

72 :param ~pacman.model.graphs.common.Slice post_vertex_slice: 

73 :param SynapseInformation synapse_info: 

74 :rtype: tuple(int, cset.connset.CSet) 

75 """ 

76 # do the work from self._cset in here 

77 

78 # this is where the magic needs to happen somehow 

79 if self.__full_cset is None: 

80 self.__full_cset = [x for x in csa.cross( 

81 range(synapse_info.n_pre_neurons), 

82 range(synapse_info.n_post_neurons)) * self.__cset] 

83 

84 # use CSA to cross the range of this vertex's neurons with the cset 

85 pair_list = csa.cross( 

86 range(pre_vertex_slice.lo_atom, pre_vertex_slice.hi_atom+1), 

87 range(post_vertex_slice.lo_atom, post_vertex_slice.hi_atom+1)) \ 

88 * self.__full_cset 

89 

90 if self.verbose: 

91 print('full cset: ', self.__full_cset) 

92 print('this vertex pair_list: ', pair_list) 

93 print('this vertex pre_neurons: ', [x[0] for x in pair_list]) 

94 print('this vertex post_neurons: ', [x[1] for x in pair_list]) 

95 

96 n_connections = len(pair_list) # size of the array created 

97 return n_connections, pair_list 

98 

99 @overrides(AbstractConnector.get_n_connections_from_pre_vertex_maximum) 

100 def get_n_connections_from_pre_vertex_maximum( 

101 self, post_vertex_slice, synapse_info, min_delay=None, 

102 max_delay=None): 

103 n_connections_max = post_vertex_slice.n_atoms 

104 

105 return self._get_n_connections_from_pre_vertex_with_delay_maximum( 

106 synapse_info.delays, 

107 synapse_info.n_pre_neurons * synapse_info.n_post_neurons, 

108 n_connections_max, min_delay, max_delay) 

109 

110 @overrides(AbstractConnector.get_n_connections_to_post_vertex_maximum) 

111 def get_n_connections_to_post_vertex_maximum(self, synapse_info): 

112 n_connections_max = synapse_info.n_pre_neurons 

113 return n_connections_max 

114 

115 @overrides(AbstractConnector.get_weight_maximum) 

116 def get_weight_maximum(self, synapse_info): 

117 n_conns_max = synapse_info.n_pre_neurons * synapse_info.n_post_neurons 

118 return self._get_weight_maximum(synapse_info.weights, n_conns_max) 

119 

120 @overrides(AbstractConnector.create_synaptic_block) 

121 def create_synaptic_block( 

122 self, pre_slices, post_slices, pre_vertex_slice, post_vertex_slice, 

123 synapse_type, synapse_info): 

124 n_connections, pair_list = self._get_n_connections( 

125 pre_vertex_slice, post_vertex_slice, synapse_info) 

126 

127 # Use whatever has been set up in _get_n_connections here 

128 # to send into the block structure 

129 # TO DO: not sure this works with repeated connector, but 

130 # what does a repeated connector mean in this context? 

131 if self.__full_connection_set is None: 

132 self.__full_connection_set = [x for x in pair_list] 

133 else: 

134 self.__full_connection_set += [x for x in pair_list] 

135 

136 block = numpy.zeros( 

137 n_connections, dtype=AbstractConnector.NUMPY_SYNAPSES_DTYPE) 

138 # source and target are the pre_neurons and post_neurons in pair_list 

139 block["source"] = [x[0] for x in pair_list] 

140 block["target"] = [x[1] for x in pair_list] 

141 block["weight"] = self._generate_weights( 

142 n_connections, None, pre_vertex_slice, post_vertex_slice, 

143 synapse_info) 

144 block["delay"] = self._generate_delays( 

145 n_connections, None, pre_vertex_slice, post_vertex_slice, 

146 synapse_info) 

147 block["synapse_type"] = synapse_type 

148 return block
