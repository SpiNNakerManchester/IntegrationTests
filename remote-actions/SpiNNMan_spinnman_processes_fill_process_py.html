65class FillProcess(AbstractMultiConnectionProcess): 

66 """ A process for filling memory. 

67 """ 

68 __slots__ = [] 

69 

70 _PACKS = [struct.Struct("<{}B".format(i)) for i in range(ALIGNMENT)] 

71 

72 # pylint: disable=too-many-arguments 

73 

74 def _write_pre_bytes(self, x, y, base, data_to_fill, size): 

75 extra_bytes = (ALIGNMENT - base % ALIGNMENT) % ALIGNMENT 

76 if not extra_bytes: 

77 return 0 

78 # Pre_bytes is the first part of the data up to the first aligned 

79 # word 

80 pre_bytes = self._PACKS[extra_bytes].pack(*data_to_fill[:extra_bytes]) 

81 # Send the pre-data to make the memory aligned (or all the 

82 # data if the size is correct - note that pre_bytes[:size] will 

83 # return all of pre_bytes if size >= len(pre_bytes) 

84 pre_data = pre_bytes[:size] 

85 if not pre_data: 

86 return 0 

87 self._send_request(WriteMemory(x, y, base, pre_data)) 

88 return extra_bytes 

89 

90 def _write_fill(self, x, y, address, base, data_to_fill, size): 

91 extra_bytes = (ALIGNMENT - base % ALIGNMENT) % ALIGNMENT 

92 size = size - ALIGNMENT if extra_bytes else size 

93 if not size: 

94 return 0 

95 # The data to send is the repeated fill data, from the end of the 

96 # pre-data circling round to the start of the post-data; we double 

97 # it up so that we don't need to use mod (it's pretty small). 

98 data = data_to_fill + data_to_fill 

99 fill_word = FillDataType.WORD.struct.unpack( 

100 data[extra_bytes:extra_bytes + ALIGNMENT - 1])[0] 

101 self._send_request(FillRequest(x, y, address, fill_word, size)) 

102 return size 

103 

104 def _write_post_bytes( 

105 self, x, y, address, base, data_to_fill, bytes_to_write): 

106 # Post bytes is the last part of the data from the end of the last 

107 # aligned word; the number of bytes to write here is exactly the 

108 # number of bytes later than a word boundary the initial address is. 

109 n_bytes = base % ALIGNMENT 

110 if not n_bytes or not bytes_to_write: 

111 return 

112 

113 post_bytes = self._PACKS[n_bytes].pack(*data_to_fill[-n_bytes:]) 

114 self._send_request(WriteMemory( 

115 x, y, address, post_bytes[:bytes_to_write]))

117 def fill_memory(self, x, y, base_address, data, size, data_type): 

118 """ 

119 :param int x: 

120 :param int y: 

121 :param int base_address: 

122 :param int data: 

123 :param int size: 

124 :param FillDataType data_type: 

125 """ 

126 # Don't do anything if there is nothing to do! 

127 if size == 0: 

128 return 

129 

130 # Check that the data can fill the requested size 

131 if size % data_type.value: 

132 raise Exception( 

133 "The size of {} bytes to fill is not divisible by the size of" 

134 " the data of {} bytes".format(size, data_type.value)) 

135 if base_address % ALIGNMENT: 

136 logger.warning( 

137 "Unaligned fill starting at %d; please use aligned fills", 

138 base_address) 

139 

140 # Get a word of data regardless of the type 

141 data_to_fill = bytearray(data_type.struct.pack( 

142 *([data] * (ALIGNMENT / data_type.value)))) 

143 

144 written = 0 

145 address = base_address 

146 

147 # Send the pre-data to make the memory aligned (or all the 

148 # data if the size is correct - note that pre_bytes[:size] will 

149 # return all of pre_bytes if size >= len(pre_bytes) 

150 delta = self._write_pre_bytes(x, y, base_address, data_to_fill, size) 

151 written += delta 

152 address += delta 

153 

154 # Fill as much as possible 

155 delta = self._write_fill( 

156 x, y, address, base_address, data_to_fill, size) 

157 written += delta 

158 address += delta 

159 

160 # Post bytes is the last part of the data from the end of the last 

161 # aligned word; send them if required 

162 self._write_post_bytes( 

163 x, y, address, base_address, data_to_fill, size - written) 

164 

165 # Wait for all the packets to be confirmed and then check there 

166 # are no errors 

167 self._finish() 

168 self.check_for_error()
