85def locate_memory_region_for_placement(placement, region, transceiver): 

86 """ Get the address of a region for a placement.

89 :param ~pacman.model.placements.Placement placement: 

90 the placement object to get the region address of 

91 :param ~spinnman.transceiver.Transceiver transceiver:

97 placement.x, placement.y, placement.p).user[0] 

98

105 placement.x, placement.y, region_offset, BYTES_PER_WORD)

215def generate_unique_folder_name(folder, filename, extension): 

216 """ Generate a unique file name with a given extension in a given folder 

217 

218 :param str folder: where to put this unique file 

219 :param str filename: 

220 the name of the first part of the file without extension 

221 :param str extension: extension of the file 

222 :return: file path with a unique addition 

223 :rtype: str 

224 """ 

225 new_file_path = os.path.join(folder, "{}{}".format(filename, extension)) 

226 count = 2 

227 while os.path.exists(new_file_path): 

228 new_file_path = os.path.join( 

229 folder, "{}_{}{}".format(filename, count, extension)) 

230 count += 1 

231 return new_file_path

263def determine_flow_states(executable_types, no_sync_changes): 

264 """ Get the start and end states for these executable types. 

265 

266 :param dict(ExecutableType,any) executable_types: 

267 the execute types to locate start and end states from 

268 :param int no_sync_changes: the number of times sync signals been sent 

269 :return: dict of executable type to states. 

270 :rtype: tuple(dict(ExecutableType,~spinnman.model.enums.CPUState),\ 

271 dict(ExecutableType,~spinnman.model.enums.CPUState)) 

272 """ 

273 expected_start_states = dict() 

274 expected_end_states = dict() 

275 for start_type in executable_types.keys(): 

276 

277 # cores that ignore all control and are just running 

278 if start_type == ExecutableType.RUNNING: 

279 expected_start_states[ExecutableType.RUNNING] = [ 

280 CPUState.RUNNING, CPUState.FINISHED] 

281 expected_end_states[ExecutableType.RUNNING] = [ 

282 CPUState.RUNNING, CPUState.FINISHED] 

283 

284 # cores that require a sync barrier 

285 elif start_type == ExecutableType.SYNC: 

286 expected_start_states[ExecutableType.SYNC] = [CPUState.SYNC0] 

287 expected_end_states[ExecutableType.SYNC] = [CPUState.FINISHED] 

288 

289 # cores that use our sim interface 

290 elif start_type == ExecutableType.USES_SIMULATION_INTERFACE: 

291 if no_sync_changes % 2 == 0: 

292 expected_start_states[start_type] = [CPUState.SYNC0] 

293 else: 

294 expected_start_states[start_type] = [CPUState.SYNC1] 

295 expected_end_states[start_type] = [CPUState.PAUSED] 

296 

297 # if no states, go boom. 

298 if not expected_start_states: 

299 raise ConfigurationException( 

300 "Unknown executable start types {}".format(executable_types)) 

301 return expected_start_states, expected_end_states

332def _emergency_state_check(txrx, app_id): 

333 """ 

334 :param ~.Transceiver txrx: 

335 :param int app_id: 

336 """ 

337 # pylint: disable=broad-except 

338 try: 

339 rte_count = txrx.get_core_state_count( 

340 app_id, CPUState.RUN_TIME_EXCEPTION) 

341 watchdog_count = txrx.get_core_state_count(app_id, CPUState.WATCHDOG) 

342 if rte_count or watchdog_count: 

343 logger.warning( 

344 "unexpected core states (rte={}, wdog={})", 

345 txrx.get_cores_in_state(None, CPUState.RUN_TIME_EXCEPTION), 

346 txrx.get_cores_in_state(None, CPUState.WATCHDOG)) 

347 except Exception: 

348 logger.exception( 

349 "Could not read the status count - going to individual cores") 

350 machine = txrx.get_machine_details() 

351 infos = CPUInfos() 

352 errors = list() 

353 for chip in machine.chips: 

354 for p in chip.processors: 

355 try: 

356 info = txrx.get_cpu_information_from_core( 

357 chip.x, chip.y, p) 

358 if info.state in ( 

359 CPUState.RUN_TIME_EXCEPTION, CPUState.WATCHDOG): 

360 infos.add_processor(chip.x, chip.y, p, info) 

361 except Exception: 

362 errors.append((chip.x, chip.y, p)) 

363 logger.warning(txrx.get_core_status_string(infos)) 

364 logger.warning("Could not read information from cores {}".format( 

365 errors))
