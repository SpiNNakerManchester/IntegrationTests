16""" The spalloc command-line tool and Python library determine their default 

17configuration options from a spalloc configuration file if present. 

18 

19.. note:: 

20 

21 Use of spalloc's configuration files is entirely optional as all 

22 configuration options may be presented as arguments to commands/methods at 

23 runtime. 

24 

25By default, configuration files are read (in ascending order of priority) from 

26a system-wide configuration directory (e.g. ``/etc/xdg/spalloc``), user 

27configuration file (e.g. ``$HOME/.config/spalloc``) and finally the current 

28working directory (in a file named ``.spalloc``). The default search paths on 

29your system can be discovered by running:: 

30 

31 $ python -m spalloc.config 

32 

33Config files use the Python :py:mod:`configparser` INI-format with a single 

34section, ``spalloc``, like so:: 

35 

36 [spalloc] 

37 hostname = localhost 

38 owner = jdh@cs.man.ac.uk 

39 

40Though most users will only wish to specify the ``hostname`` and ``owner`` 

41options (as in the example above), the following enumerates the complete set of 

42options available (and the default value). 

43 

44``hostname`` 

45 The hostname or IP address of the spalloc-server to connect to. 

46``owner`` 

47 The name of the owner of created jobs. By convention the user's email 

48 address. 

49``port`` 

50 The port used by the spalloc-server. (Default: 22244) 

51``keepalive`` 

52 The keepalive interval, in seconds, to use when creating jobs. If the 

53 spalloc-server does not receive a keepalive command for this interval the 

54 job is automatically destroyed. May be set to None to disable this feature. 

55 (Default: 60.0) 

56``reconnect_delay`` 

57 The time, in seconds, to wait between reconnection attempts to the server 

58 if disconnected. (Default 5.0) 

59``timeout`` 

60 The time, in seconds, to wait before giving up waiting for a response from 

61 the server or None to wait forever. (Default 5.0) 

62``machine`` 

63 The name of a specific machine on which to run all jobs or None to use any 

64 available machine. (Default: None) 

65``tags`` 

66 The set of tags, comma seperated, to require a machine to have when 

67 allocating jobs. (Default: default) 

68``min_ratio`` 

69 Require that when allocating a number of boards the allocation is at least 

70 as square as this aspect ratio. (Default: 0.333) 

71``max_dead_boards`` 

72 The maximum number of dead boards which may be present in an allocated set 

73 of boards or None to allow any number of dead boards. (Default: 0) 

74``max_dead_links`` 

75 The maximum number of dead links which may be present in an allocated set 

76 of boards or None to allow any number of dead links. (Default: None) 

77``require_torus`` 

78 If True, require that an allocation have wrap-around links. This typically 

79 requires the allocation of a whole machine. If False, wrap-around links may 

80 or may-not be present in allocated machines. (Default: False) 

81"""

142def read_config(filenames=None): 

143 """ Attempt to read local configuration files to determine spalloc client 

144 settings. 

145 

146 Parameters 

147 ---------- 

148 filenames : [str, ...] 

149 Filenames to attempt to read. Later config file have higher priority. 

150 

151 Returns 

152 ------- 

153 dict 

154 The configuration loaded. 

155 """ 

156 if filenames is None: # pragma: no cover 

157 filenames = SEARCH_PATH 

158 parser = configparser.ConfigParser() 

159 

160 # Set default config values (NB: No read_dict in Python 2.7) 

161 parser.add_section(SECTION) 

162 for key, value in iteritems(DEFAULT_CONFIG): 

163 parser.set(SECTION, key, value) 

164 

165 # Attempt to read from each possible file location in turn 

166 for filename in filenames: 

167 try: 

168 with open(filename, "r") as f: 

169 # TODO: Remove use of readfp once we stop supporting 2.7 

170 if PY3: 

171 parser.read_file( # pylint: disable=no-member 

172 f, filename) 

173 else: 

174 parser.readfp( # pylint: disable=deprecated-method 

175 f, filename) 

176 except (IOError, OSError): 

177 # File did not exist, keep trying 

178 pass 

179 

180 cfg = { 

181 "hostname": _read_any_str(parser, "hostname"), 

182 "owner": _read_any_str(parser, "owner"), 

183 "port": parser.getint(SECTION, "port"), 

184 "keepalive": _read_none_or_float(parser, "keepalive"), 

185 "reconnect_delay": parser.getfloat(SECTION, "reconnect_delay"), 

186 "timeout": _read_none_or_float(parser, "timeout"), 

187 "machine": _read_none_or_str(parser, "machine"), 

188 "min_ratio": parser.getfloat(SECTION, "min_ratio"), 

189 "max_dead_boards": _read_none_or_int(parser, "max_dead_boards"), 

190 "max_dead_links": _read_none_or_int(parser, "max_dead_links"), 

191 "require_torus": parser.getboolean(SECTION, "require_torus"), 

192 "ignore_version": parser.getboolean(SECTION, "ignore_version")} 

193 

194 tags = _read_none_or_str(parser, "tags") 

195 cfg["tags"] = None if tags is None else list( 

196 map(str.strip, tags.split(","))) 

197 

198 return cfg
