28class HostDataRead(EIEIOCommandMessage): 

29 """ Packet sent by the host computer to the SpiNNaker system in the\ 

30 context of the buffering output technique to signal that the host has\ 

31 completed reading data from the output buffer, and that such space can\ 

32 be considered free to use again 

33 """ 

34 __slots__ = [ 

35 "_acks", 

36 "_header"] 

37 

38 def __init__( 

39 self, n_requests, sequence_no, channel, region_id, space_read): 

40 # pylint: disable=too-many-arguments 

41 if not isinstance(channel, list): 

42 channel = [channel] 

43 

44 if not isinstance(region_id, list): 

45 region_id = [region_id] 

46 

47 if not isinstance(space_read, list): 

48 space_read = [space_read] 

49 

50 if len(channel) != n_requests or \ 

51 len(region_id) != n_requests or \ 

52 len(space_read) != n_requests: 

53 raise SpinnmanInvalidPacketException( 

54 "SpinnakerRequestReadData", 

55 "The format for a SpinnakerRequestReadData packet is " 

56 "invalid: {0:d} request(s), {1:d} space(s) read " 

57 "defined, {2:d} region(s) defined, {3:d} channel(s) " 

58 "defined".format( 

59 n_requests, len(space_read), len(region_id), len(channel))) 

60 super(HostDataRead, self).__init__(EIEIOCommandHeader( 

61 EIEIO_COMMAND_IDS.HOST_DATA_READ)) 

62 self._header = _HostDataReadHeader(n_requests, sequence_no) 

63 self._acks = _HostDataReadAck(channel, region_id, space_read)

82 @staticmethod 

83 def get_min_packet_length(): 

84 return 8

86 @staticmethod 

87 def from_bytestring(command_header, data, offset): # @UnusedVariable 

88 n_requests, sequence_no = _PATTERN_BB.unpack_from(data, offset) 

89 

90 offset += 2 

91 n_requests &= 0x7 

92 channel = list() 

93 region_id = list() 

94 space_read = list() 

95 

96 for _ in xrange(n_requests): 

97 channel_ack, region_id_ack, space_read_ack = \ 

98 _PATTERN_xxBBI.unpack_from(data, offset) 

99 channel.append(channel_ack) 

100 region_id.append(region_id_ack) 

101 space_read.append(space_read_ack) 

102 offset += 8 

103 return HostDataRead( 

104 n_requests, sequence_no, channel, region_id, space_read)

106 @property 

107 def bytestring(self): 

108 byte_string = super(HostDataRead, self).bytestring 

109 n_requests = self.n_requests 

110 byte_string += _PATTERN_BB.pack(n_requests, self.sequence_no) 

111 for i in xrange(n_requests): 

112 byte_string += _PATTERN_xxBBI.pack( 

113 self.channel(i), self.region_id(i), self.space_read(i)) 

114 return byte_string
