54 def create_aggregated_citation_file( 

55 self, module_to_start_at, aggregated_citation_file): 

56 """ Entrance method for building the aggregated citation file 

57 

58 :param module_to_start_at:\ 

59 the top level module to figure out its citation file for 

60 :type module_to_start_at: python module 

61 :param aggregated_citation_file: file name of aggregated citation file 

62 :type file_path_of_aggregated_citation_file: str 

63 :rtype: None 

64 """ 

65 

66 # get the top citation file to add references to 

67 top_citation_file_path = os.path.join(os.path.dirname(os.path.dirname( 

68 os.path.abspath(module_to_start_at.__file__))), CITATION_FILE) 

69 modules_seen_so_far = set() 

70 modules_seen_so_far.add("") # Make sure the empty entry is absent 

71 with open(top_citation_file_path, 'r') as stream: 

72 top_citation_file = yaml.safe_load(stream) 

73 top_citation_file[REFERENCES_YAML_POINTER] = list() 

74 

75 # get the dependency list 

76 requirements_file_path = os.path.join( 

77 os.path.dirname(os.path.dirname(os.path.abspath( 

78 module_to_start_at.__file__))), REQUIREMENTS_FILE) 

79 c_requirements_file_path = os.path.join( 

80 os.path.dirname(os.path.dirname(os.path.abspath( 

81 module_to_start_at.__file__))), C_REQUIREMENTS_FILE) 

82 

83 # attempt to get python PYPI to import command map 

84 pypi_to_import_map_file = os.path.join( 

85 os.path.dirname(os.path.dirname(os.path.abspath( 

86 module_to_start_at.__file__))), 

87 PYPI_TO_IMPORT_FILE) 

88 pypi_to_import_map = None 

89 if os.path.isfile(pypi_to_import_map_file): 

90 pypi_to_import_map = self._read_pypi_import_map( 

91 pypi_to_import_map_file) 

92 

93 if os.path.isfile(requirements_file_path): 

94 with open(requirements_file_path, "r") as r_file: 

95 for line in r_file: 

96 module = line.strip() 

97 if module.startswith("#"): 

98 continue 

99 if module not in modules_seen_so_far: 

100 import_name = pypi_to_import_map.get(module, module) 

101 # pylint: disable=broad-except 

102 try: 

103 imported_module = importlib.import_module( 

104 import_name) 

105 self._handle_python_dependency( 

106 top_citation_file, imported_module, 

107 modules_seen_so_far, 

108 pypi_to_import_map[module]) 

109 except Exception as e: # pragma: no cover 

110 print("Error handling python dependency {}: {}" 

111 .format(module, str(e))) 

112 

113 if os.path.isfile(c_requirements_file_path): 

114 with open(c_requirements_file_path, "r") as r_file: 

115 for line in r_file: 

116 module = line.strip() 

117 if module.startswith("#"): 

118 continue 

119 if module not in modules_seen_so_far: 

120 self._handle_c_dependency( 

121 top_citation_file, module, modules_seen_so_far) 

122 

123 # write citation file with updated fields 

124 with io.open( 

125 aggregated_citation_file, 'w', encoding='utf8') as outfile: 

126 yaml.dump(top_citation_file, outfile, default_flow_style=False, 

127 allow_unicode=True)

129 @staticmethod 

130 def _read_pypi_import_map(aggregated_citation_file): 

131 """ Read the PYPI to import name map 

132 

133 :param aggregated_citation_file: file path to the PYPI map 

134 :return: map between PYPI names and import names 

135 """ 

136 pypi_to_import_map = dict() 

137 for line in open(aggregated_citation_file, "r"): 

138 [pypi, import_command] = line.split(":") 

139 pypi_to_import_map[pypi] = import_command.split("\n")[0] 

140 return pypi_to_import_map

142 def _handle_c_dependency( 

143 self, top_citation_file, module, modules_seen_so_far): 

144 """ Handle a c code dependency 

145 

146 :param top_citation_file: YAML file for the top citation file 

147 :param module: module to find 

148 :type top_citation_file: YAML file 

149 :type module: str 

150 :return: None 

151 """ 

152 cleaned_path = self.locate_path_for_c_dependency(module) 

153 if cleaned_path is not None: 

154 # process reference 

155 reference_entry = self._process_reference( 

156 cleaned_path, None, modules_seen_so_far, module) 

157 

158 # append to the top citation file 

159 top_citation_file[REFERENCES_YAML_POINTER].append( 

160 reference_entry) 

161 self._search_for_other_c_references( 

162 reference_entry, cleaned_path, modules_seen_so_far) 

163 else: 

164 print("Could not find C dependency {}".format(module))

166 @staticmethod 

167 def locate_path_for_c_dependency(true_software_name): 

168 environment_path_variable = os.environ.get('PATH') 

169 if environment_path_variable is not None: 

170 software_paths = environment_path_variable.split(":") 

171 for software_path in software_paths: 

172 # clear path to have repository name at end 

173 last_version = None 

174 cleaned_path = software_path 

175 while ((cleaned_path != last_version) and ( 

176 not (cleaned_path.split(os.sep)[-1] == 

177 true_software_name))): 

178 last_version = cleaned_path 

179 cleaned_path = os.path.dirname(cleaned_path) 

180 

181 if cleaned_path != last_version: 

182 return cleaned_path 

183 return None

185 def _search_for_other_c_references( 

186 self, reference_entry, software_path, modules_seen_so_far): 

187 """ Go though the top level path and tries to locate other cff \ 

188 files that need to be added to the references pile 

189 

190 :param reference_entry:\ 

191 The reference entry to add new dependencies as references for. 

192 :param software_path: the path to search in 

193 :rtype: None 

194 """ 

195 for possible_extra_citation_file in os.listdir(software_path): 

196 if possible_extra_citation_file.endswith(".cff"): 

197 dependency_reference_entry = \ 

198 self._read_and_process_reference_entry( 

199 os.path.join(software_path, 

200 possible_extra_citation_file)) 

201 reference_entry[REFERENCES_YAML_POINTER] = list() 

202 reference_entry[REFERENCES_YAML_POINTER].append( 

203 dependency_reference_entry) 

204 modules_seen_so_far.add( 

205 possible_extra_citation_file.split(".")[0])

207 def _handle_python_dependency( 

208 self, top_citation_file, imported_module, modules_seen_so_far, 

209 module_name): 

210 """ Handle a python dependency 

211 

212 :param top_citation_file: YAML file for the top citation file 

213 :type top_citation_file: YAML file 

214 :param imported_module: the actual imported module 

215 :type imported_module: ModuleType 

216 :param modules_seen_so_far: \ 

217 list of names of dependencies already processed 

218 :type modules_seen_so_far: list 

219 :param module_name: the name of this module to consider as a dependency 

220 :type module_name: str 

221 :rtype: None 

222 """ 

223 # get modules citation file 

224 citation_level_dir = os.path.abspath(imported_module.__file__) 

225 m_path = module_name.replace(".", os.sep) 

226 last_citation_level_dir = None 

227 while (not citation_level_dir.endswith(m_path) and 

228 last_citation_level_dir != citation_level_dir): 

229 last_citation_level_dir = citation_level_dir 

230 citation_level_dir = os.path.dirname(citation_level_dir) 

231 if citation_level_dir == last_citation_level_dir: # pragma: no cover 

232 raise Exception("Folder for module {} not found".format( 

233 module_name)) 

234 

235 # get the reference data for the reference 

236 reference_entry = self._process_reference( 

237 citation_level_dir, imported_module, modules_seen_so_far, 

238 module_name) 

239 

240 if reference_entry is not None: 

241 # append to the top citation file 

242 top_citation_file[REFERENCES_YAML_POINTER].append(reference_entry)

244 def _process_reference( 

245 self, citation_level_dir, imported_module, modules_seen_so_far, 

246 module_name): 

247 """ Take a module level and tries to locate and process a citation file 

248 

249 :param citation_level_dir: \ 

250 the expected level where the CITATION.cff should be 

251 :type citation_level_dir: str 

252 :param imported_module: the module after being imported 

253 :type imported_module: python module 

254 :param modules_seen_so_far: list of dependencies already processed 

255 :type modules_seen_so_far: list 

256 :return: the reference entry in json format 

257 :rtype: dict 

258 """ 

259 

260 # if it exists, add it as a reference to the top one 

261 if os.path.isfile(os.path.join(citation_level_dir, CITATION_FILE)): 

262 reference_entry = self._read_and_process_reference_entry( 

263 os.path.join(citation_level_dir, CITATION_FILE)) 

264 

265 # check that the file isn't one above (not installed, but developer 

266 # mode) 

267 elif os.path.isfile(os.path.join(os.path.dirname( 

268 os.path.abspath(citation_level_dir)), CITATION_FILE)): 

269 reference_entry = self._read_and_process_reference_entry( 

270 os.path.join(os.path.dirname( 

271 os.path.abspath(citation_level_dir)), CITATION_FILE)) 

272 

273 # if no citation file exists, do an attempt to find a version to build 

274 # from 

275 else: 

276 # one from version 

277 reference_entry = self._try_to_find_version( 

278 imported_module, module_name) 

279 

280 modules_seen_so_far.add(imported_module) 

281 return reference_entry

283 @staticmethod 

284 def _try_to_find_version(imported_module, module_name): 

285 """ Try to locate a version file or version data to auto-generate \ 

286 minimal citation data. 

287 

288 :param imported_module:\ 

289 the module currently trying to find the version of 

290 :type imported_module: python module 

291 :return: reference entry for this python module 

292 :rtype: dict 

293 """ 

294 reference_entry = dict() 

295 reference_entry[REFERENCES_TYPE_TYPE] = REFERENCES_SOFTWARE_TYPE 

296 reference_entry[REFERENCES_TITLE_TYPE] = module_name 

297 if (hasattr(imported_module, "__version_day__") and 

298 hasattr(imported_module, "__version_month__") and 

299 hasattr(imported_module, "__version_year__")): 

300 reference_entry[REFERENCES_DATE_TYPE] = \ 

301 CitationUpdaterAndDoiGenerator.\ 

302 convert_text_date_to_date( 

303 version_day=imported_module.__version_day__, 

304 version_month=imported_module.__version_month__, 

305 version_year=imported_module.__version_year__) 

306 if hasattr(imported_module, "__version__"): 

307 reference_entry[REFERENCES_VERSION_TYPE] = \ 

308 imported_module.__version__ 

309 elif hasattr(imported_module, "version"): 

310 reference_entry[REFERENCES_VERSION_TYPE] = \ 

311 imported_module.version 

312 elif hasattr(imported_module, "_version"): 

313 reference_entry[REFERENCES_VERSION_TYPE] = \ 

314 imported_module._version 

315 return reference_entry

317 @staticmethod 

318 def _read_and_process_reference_entry(dependency_citation_file_path): 

319 """ Read a CITATION.cff and makes it a reference for a higher level \ 

320 citation file. 

321 

322 :param dependency_citation_file_path: path to a CITATION.cff file 

323 :type dependency_citation_file_path: str 

324 :return: reference entry for the higher level citation.cff 

325 :rtype: dict 

326 """ 

327 reference_entry = dict() 

328 

329 with open(dependency_citation_file_path, 'r') as stream: 

330 dependency_citation_file = yaml.safe_load(stream) 

331 

332 reference_entry[REFERENCES_TYPE_TYPE] = REFERENCES_SOFTWARE_TYPE 

333 reference_entry[REFERENCES_AUTHORS_TYPE] = \ 

334 dependency_citation_file[REFERENCES_AUTHORS_TYPE] 

335 reference_entry[REFERENCES_TITLE_TYPE] = \ 

336 dependency_citation_file[REFERENCES_TITLE_TYPE] 

337 reference_entry[REFERENCES_CONTACT_TYPE] = \ 

338 dependency_citation_file[REFERENCES_CONTACT_TYPE] 

339 reference_entry[REFERENCES_VERSION_TYPE] = \ 

340 dependency_citation_file[REFERENCES_VERSION_TYPE] 

341 reference_entry[REFERENCES_DATE_TYPE] = \ 

342 dependency_citation_file[REFERENCES_DATE_TYPE] 

343 reference_entry[REFERENCES_URL_TYPE] = \ 

344 dependency_citation_file[REFERENCES_URL_TYPE] 

345 reference_entry[REFERENCES_REPO_TYPE] = \ 

346 dependency_citation_file[REFERENCES_REPO_TYPE] 

347 return reference_entry

350def generate_aggregate(arguments=None): 

351 """ Generates a single citation.cff from others 

352 

353 :param output_path: Where to write the aggregate file 

354 :param top_module: the module to start aggregating the citation.cffs from 

355 :param doi_title: the title of the DOI 

356 :param zenodo_access_token: the access token for Zenodo 

357 :param tools_doi: the DOI of the tools 

358 :rtype: None 

359 """ 

360 parser = argparse.ArgumentParser(description="Aggregate Citations") 

361 parser.add_argument("output_path", help="The file to store the result in") 

362 parser.add_argument("top_module", help="The module to start with") 

363 parser.add_argument("--create_doi", action="store_true", 

364 help="Create a DOI from the resulting citation" 

365 " on Zenodo") 

366 parser.add_argument("--publish_doi", action="store_true", 

367 help="Publish the DOI created") 

368 parser.add_argument("--doi_title", 

369 help="The title to give the created DOI") 

370 parser.add_argument("--previous_doi", 

371 help="The DOI this is a newer version of") 

372 parser.add_argument("--zenodo_access_token", 

373 help="Access token for Zenodo") 

374 

375 args = parser.parse_args(arguments) 

376 error = False 

377 if args.create_doi: # pragma: no cover 

378 if not args.doi_title: 

379 print("--doi_title required when creating a DOI") 

380 error = True 

381 if not args.previous_doi: 

382 print("--previous_doi required when creating a DOI") 

383 error = True 

384 if not args.zenodo_access_token: 

385 print("--zenodo_access_token required when creating a DOI") 

386 error = True 

387 if args.publish_doi and not args.create_doi: # pragma: no cover 

388 print("Cannot publish DOI without creating one") 

389 error = True 

390 if error: # pragma: no cover 

391 parser.print_usage() 

392 sys.exit() 

393 

394 top_module = importlib.import_module(args.top_module) 

395 

396 citation_aggregator = CitationAggregator() 

397 citation_aggregator.create_aggregated_citation_file( 

398 top_module, args.output_path) 

399 citation_updater_and_dio_generator = CitationUpdaterAndDoiGenerator() 

400 citation_updater_and_dio_generator.update_citation_file_and_create_doi( 

401 citation_file_path=args.output_path, 

402 doi_title=args.doi_title, 

403 create_doi=args.create_doi, publish_doi=args.publish_doi, 

404 previous_doi=args.previous_doi, 

405 zenodo_access_token=args.zenodo_access_token, 

406 module_path=os.path.dirname(top_module.__path__[0])) 

407 

408 

409if __name__ == "__main__": 

410 generate_aggregate() # pragma: no cover
