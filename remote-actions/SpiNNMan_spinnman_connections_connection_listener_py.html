16import logging 

17from threading import Thread 

18from concurrent.futures import ThreadPoolExecutor 

19from spinn_utilities.abstract_context_manager import AbstractContextManager 

20 

21logger = logging.getLogger(__name__) 

22_POOL_SIZE = 4 

23_TIMEOUT = 1 

24 

25 

26class ConnectionListener(Thread, AbstractContextManager): 

27 """ Thread that listens to a connection and calls callbacks with new\ 

28 messages when they arrive. 

29 """ 

30 __slots__ = [ 

31 "__callback_pool", 

32 "__callbacks", 

33 "__connection", 

34 "__done", 

35 "__timeout"] 

36 

37 def __init__(self, connection, n_processes=_POOL_SIZE, timeout=_TIMEOUT): 

38 """ 

39 :param Listenable connection: A connection to listen to 

40 :param int n_processes: 

41 The number of threads to use when calling callbacks 

42 :param float timeout: 

43 How long to wait for messages before checking to see if the 

44 connection is to be terminated. 

45 """ 

46 super(ConnectionListener, self).__init__( 

47 name="Connection listener for connection {}".format(connection)) 

48 self.daemon = True 

49 self.__connection = connection 

50 self.__timeout = timeout 

51 self.__callback_pool = ThreadPoolExecutor(max_workers=n_processes) 

52 self.__done = False 

53 self.__callbacks = set() 

54 

55 def __run_step(self, handler): 

56 """ 

57 :param callable handler: 

58 """ 

59 if self.__connection.is_ready_to_receive(timeout=self.__timeout): 

60 message = handler() 

61 for callback in self.__callbacks: 

62 future = self.__callback_pool.submit(callback, message) 

63 future.add_done_callback(self.__done_callback) 

64 

65 def __done_callback(self, future): 

66 """ 

67 :param ~concurrent.futures.Future future: 

68 """ 

69 try: 

70 future.result() 

71 except Exception: # pylint: disable=broad-except 

72 logger.exception("problem in listener call") 

73 

74 def run(self): 

75 """ Implements the listening thread. 

76 """ 

77 try: 

78 handler = self.__connection.get_receive_method() 

79 while not self.__done: 

80 try: 

81 self.__run_step(handler) 

82 except Exception: # pylint: disable=broad-except 

83 if not self.__done: 

84 logger.warning("problem when dispatching message", 

85 exc_info=True) 

86 finally: 

87 self.__callback_pool.shutdown() 

88 self.__callback_pool = None 

89 

90 def add_callback(self, callback): 

91 """ Add a callback to be called when a message is received 

92 

93 :param callable callback: 

94 A callable which takes a single parameter, which is the message 

95 received; the result of the callback will be ignored. 

96 """ 

97 self.__callbacks.add(callback) 

98 

99 def close(self): 

100 """ Closes the listener. Note that this does not close the provider\ 

101 of the messages; this instead marks the listener as closed. The\ 

102 listener will not truly stop until the get message call returns. 

103 """ 

104 self.__done = True 

105 self.join()
