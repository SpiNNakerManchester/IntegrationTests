16import requests 

17from spinn_machine import Machine 

18from spinn_machine.virtual_machine import virtual_machine 

19 

20 

21class HBPMaxMachineGenerator(object): 

22 """ Generates a virtual machine of the width and height of the maximum\ 

23 machine a given HBP server can generate. 

24 

25 :param str hbp_server_url: 

26 The URL of the HBP server from which to get the machine 

27 :param int total_run_time: The total run time to request 

28 :param int max_machine_core_reduction: 

29 the number of cores less than 

30 :py:const:`~spinn_machine.Machine.DEFAULT_MAX_CORES_PER_CHIP` 

31 that each chip should have 

32 :rtype: ~spinn_machine.Machine 

33 """ 

34 

35 __slots__ = [] 

36 

37 def __call__(self, hbp_server_url, total_run_time, 

38 max_machine_core_reduction=0): 

39 """ 

40 :param str hbp_server_url: 

41 :param int total_run_time: 

42 :param int max_machine_core_reduction: 

43 :rtype: ~.Machine 

44 """ 

45 

46 max_machine = self._max_machine_request(hbp_server_url, total_run_time) 

47 

48 n_cpus_per_chip = (Machine.max_cores_per_chip() - 

49 max_machine_core_reduction) 

50 

51 # Return the width and height and assume that it has wrap arounds 

52 return virtual_machine( 

53 width=max_machine["width"], height=max_machine["height"], 

54 n_cpus_per_chip=n_cpus_per_chip, validate=False) 

55 

56 def _max_machine_request(self, url, total_run_time): 

57 """ 

58 :param str url: 

59 :param int total_run_time: 

60 :rtype: dict 

61 """ 

62 if url.endswith("/"): 

63 url = url[:-1] 

64 r = requests.get("{}/max".format(url), params={ 

65 'runTime': total_run_time}) 

66 r.raise_for_status() 

67 return r.json()
