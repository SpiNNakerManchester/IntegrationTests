51def compute_energy_used(checkpoint: Optional[int] = None) -> PowerUsed: 

52 """ 

53 This algorithm does the actual work of computing energy used by a 

54 simulation (or other application) running on SpiNNaker. 

55 

56 :param checkpoint: the time at which to compute execution energy up to 

57 """ 

58 # Get data from provenance 

59 with GlobalProvenance() as db: 

60 waiting_ms = db.get_category_timer_sum_by_reset(TimerCategory.WAITING) 

61 setup_ms = db.get_timer_sum_by_category_and_reset( 

62 TimerCategory.SETTING_UP) 

63 get_machine_ms = db.get_timer_sum_by_category_and_reset( 

64 TimerCategory.GET_MACHINE) 

65 

66 mapping_ms = db.get_timer_sum_by_category_and_reset( 

67 TimerCategory.MAPPING) 

68 loading_ms = db.get_timer_sum_by_category_and_reset( 

69 TimerCategory.LOADING) 

70 

71 run_other_ms = db.get_timer_sum_by_category_and_reset( 

72 TimerCategory.RUN_OTHER) 

73 run_loop_ms = db.get_timer_sum_by_category_and_reset( 

74 TimerCategory.RUN_LOOP) 

75 resetting_ms = db.get_timer_sum_by_category_and_reset( 

76 TimerCategory.RESETTING) 

77 

78 shutting_down_ms = db.get_timer_sum_by_category_and_reset( 

79 TimerCategory.SHUTTING_DOWN) 

80 

81 # Separate out processes that are part of the others but that happen 

82 # on the machine, so we can account for active machine, not idle 

83 data_loading_ms = db.get_timer_sum_by_work(TimerWork.LOADING_DATA) 

84 loading_ms -= data_loading_ms 

85 data_extraction_ms = 0 

86 if get_config_bool("Machine", "enable_advanced_monitor_support"): 

87 data_extraction_ms = db.get_timer_sum_by_work( 

88 TimerWork.EXTRACT_DATA) 

89 run_loop_ms -= data_extraction_ms 

90 expansion_ms = db.get_timer_sum_by_work(TimerWork.SYNAPSE) 

91 loading_ms -= expansion_ms 

92 

93 if checkpoint is not None: 

94 execute_on_machine_ms = checkpoint 

95 else: 

96 timesteps = FecDataView.get_current_run_timesteps() 

97 if timesteps is None: 

98 raise ValueError( 

99 "Cannot compute energy without knowing the end time") 

100 ts_factor = FecDataView.get_time_scale_factor() 

101 execute_on_machine_ms = int(round(timesteps * ts_factor)) 

102 

103 run_loop_ms -= execute_on_machine_ms 

104 

105 machine = FecDataView.get_machine() 

106 version = FecDataView.get_machine_version() 

107 n_boards = len(machine.ethernet_connected_chips) 

108 n_chips = machine.n_chips 

109 n_cores = FecDataView.get_n_placements() 

110 n_frames = _calculate_n_frames(machine) 

111 

112 active_cores: Dict[Tuple[int, int], int] = defaultdict(int) 

113 power_cores: Dict[Tuple[int, int], int] = {} 

114 n_active_cores = 0 

115 for pl in FecDataView.iterate_placemements(): 

116 if not isinstance(pl.vertex, AbstractHasAssociatedBinary): 

117 continue 

118 vertex: AbstractHasAssociatedBinary = cast( 

119 AbstractHasAssociatedBinary, pl.vertex) 

120 if vertex.get_binary_start_type() != ExecutableType.SYSTEM: 

121 if isinstance(vertex, ChipPowerMonitorMachineVertex): 

122 power_cores[(pl.x, pl.y)] = pl.p 

123 else: 

124 active_cores[(pl.x, pl.y)] += 1 

125 n_active_cores += 1 

126 n_active_chips = len(active_cores) 

127 

128 # TODO confirm Power monitors are not included here 

129 extra_monitors_per_chip = (version.n_scamp_cores 

130 + FecDataView.get_all_monitor_cores() - 1) 

131 extra_monitors_per_board = (version.n_scamp_cores + 

132 FecDataView.get_ethernet_monitor_cores() - 1) 

133 if get_config_bool("Reports", "write_energy_report"): 

134 run_chip_active_time = _extract_cores_active_time( 

135 checkpoint, active_cores, power_cores, version) 

136 else: 

137 run_chip_active_time = _assume_core_always_active( 

138 active_cores, execute_on_machine_ms) 

139 load_chip_active_time = _make_extra_monitor_core_use( 

140 data_loading_ms, machine, extra_monitors_per_board, 

141 extra_monitors_per_chip) 

142 extraction_chip_active_time = _make_extra_monitor_core_use( 

143 data_extraction_ms, machine, extra_monitors_per_board, 

144 extra_monitors_per_chip) 

145 

146 run_router_packets = _extract_router_packets("Run", version) 

147 load_router_packets = _extract_router_packets("Load", version) 

148 extraction_router_packets = _extract_router_packets("Extract", version) 

149 

150 # TODO get_machine not include here 

151 return compute_energy_over_time( 

152 waiting_ms, setup_ms, get_machine_ms, mapping_ms, loading_ms, 

153 data_loading_ms, expansion_ms, data_extraction_ms, run_other_ms, 

154 run_loop_ms, execute_on_machine_ms, resetting_ms, shutting_down_ms, 

155 version, n_chips, n_active_chips, n_boards, n_frames, n_cores, 

156 n_active_cores, load_chip_active_time, extraction_chip_active_time, 

157 run_chip_active_time, load_router_packets, extraction_router_packets, 

158 run_router_packets)

161def _calculate_n_frames(machine: Machine) -> int: 

162 """ 

163 Figures out how many frames are being used in this setup. 

164 A key of cabinet,frame will be used to identify unique frame. 

165 

166 :param machine: the machine object 

167 :return: number of frames 

168 """ 

169 # if not spalloc, then could be any type of board, but unknown cooling 

170 if not FecDataView.has_allocation_controller(): 

171 return 0 

172 

173 # using spalloc in some form; how many unique frames? 

174 cabinet_frame = set() 

175 mac = FecDataView.get_allocation_controller() 

176 for ethernet_connected_chip in machine.ethernet_connected_chips: 

177 cabinet, frame, _ = mac.where_is_machine( 

178 ethernet_connected_chip.x, ethernet_connected_chip.y) 

179 cabinet_frame.add((cabinet, frame)) 

180 return len(cabinet_frame)

183def _extract_router_packets( 

184 prefix: str, version: AbstractVersion) -> RouterPackets: 

185 packets_per_chip: RouterPackets = defaultdict(dict) 

186 with ProvenanceReader() as db: 

187 for name in version.get_router_report_packet_types(): 

188 for (x, y, value) in db.get_router_by_chip(f"{prefix}{name}"): 

189 packets_per_chip[x, y][name] = value 

190 return packets_per_chip

193def _extract_cores_active_time( 

194 checkpoint: Optional[int], active_cores: Dict[Tuple[int, int], int], 

195 power_cores: Dict[Tuple[int, int], int], 

196 version: AbstractVersion) -> ChipActiveTime: 

197 sampling_frequency = get_config_int("EnergyMonitor", "sampling_frequency") 

198 

199 chip_activity: ChipActiveTime = {} 

200 with BufferDatabase() as buff_db: 

201 for (x, y), n_cores in active_cores.items(): 

202 # Find the core that was used on this chip for power monitoring 

203 p = power_cores[(x, y)] 

204 # Get time per sample in seconds (frequency in microseconds) 

205 time_for_recorded_sample_s = sampling_frequency / _US_PER_SECOND 

206 data, _missing = buff_db.get_recording(x, y, p, RECORDING_CHANNEL) 

207 results = numpy.frombuffer(data, dtype=numpy.uint32).reshape( 

208 -1, version.max_cores_per_chip + 1) 

209 # Get record times in milliseconds (frequency in microseconds) 

210 record_times = results[:, 0] * sampling_frequency / _US_PER_MS 

211 # The remaining columns are the counts of active / inactive at 

212 # each sample point 

213 activity = results[:, 1:].astype(numpy.float64) 

214 # Set the activity of *this* core to 0, as we don't want to 

215 # measure that! 

216 physical_core = FecDataView.get_physical_core_id((x, y), p) 

217 activity[:, physical_core] = 0 

218 # Convert to actual active time, assuming the core is fully active 

219 # or fully inactive between samples 

220 activity_times = activity * time_for_recorded_sample_s 

221 # If checkpoint is specified, filter the times 

222 if checkpoint is not None: 

223 activity_times = activity_times[record_times < checkpoint] 

224 chip_activity[x, y] = (activity_times.sum(), n_cores) 

225 return chip_activity

228def _assume_core_always_active( 

229 active_cores: Dict[Tuple[int, int], int], 

230 execute_on_machine_ms: float) -> ChipActiveTime: 

231 """ 

232 As there are no power monitors assume cores always active 

233 

234 """ 

235 logger.warning( 

236 "Energy monitoring cores not enabled, assuming all cores were" 

237 " active for whole run time. To get a better energy estimate," 

238 " set write_energy_report=True in the [Reports] section of the" 

239 " configuration file") 

240 chip_activity: ChipActiveTime = {} 

241 for (x, y), n_cores in active_cores.items(): 

242 chip_activity[x, y] = ( 

243 (execute_on_machine_ms * n_cores) / _MS_PER_SECOND, n_cores) 

244 return chip_activity

247def _make_extra_monitor_core_use( 

248 time_ms: int, machine: Machine, extra_monitors_per_board: int, 

249 extra_monitors_per_chip: int) -> ChipActiveTime: 

250 time_s = time_ms / _MS_PER_SECOND 

251 core_use = {} 

252 for chip in machine.chips: 

253 n_monitors = extra_monitors_per_chip 

254 if chip.ip_address is not None: 

255 n_monitors += extra_monitors_per_board 

256 core_use[chip.x, chip.y] = (n_monitors * time_s, n_monitors) 

257 return core_use

260def compute_energy_over_time( 

261 waiting_ms: float, setup_ms: float, get_machine_ms: float, 

262 mapping_ms: float, loading_ms: float, data_loading_ms: float, 

263 expansion_ms: float, data_extraction_ms: float, run_other_ms: float, 

264 run_loop_ms: float, execute_on_machine_ms: float, 

265 resetting_ms: float, shutting_down_ms: float, version: AbstractVersion, 

266 n_chips: int, n_active_chips: int, n_boards: int, n_frames: int, 

267 n_cores: int, n_active_cores: int, 

268 load_chip_active_time: ChipActiveTime, 

269 extraction_chip_active_time: ChipActiveTime, 

270 run_chip_active_time: ChipActiveTime, 

271 load_router_packets: RouterPackets, 

272 extraction_router_packets: RouterPackets, 

273 run_router_packets: RouterPackets) -> PowerUsed: 

274 """ 

275 Compute the energy used by a simulation running on SpiNNaker. 

276 

277 :param waiting_ms: time spent waiting for things to happen in general 

278 :param setup_ms: time spent setting up the simulation 

279 :param get_machine_ms: time spent getting the machine 

280 :param mapping_ms: time spent mapping to the machine 

281 :param loading_ms: 

282 time spent loading the simulation onto the passive machine 

283 :param data_loading_ms: 

284 time spent loading data onto the machine actively using the machine to 

285 load the data 

286 :param expansion_ms: time spent expanding the data on the machine 

287 :param data_extraction_ms: 

288 time spent extracting data from the machine actively using the machine 

289 to extract the data 

290 :param run_other_ms: 

291 time spent in running but not the active machine time, just in between 

292 calls to other things 

293 :param run_loop_ms: 

294 time spent in running but not the active machine time, just in the 

295 run loop itself 

296 :param execute_on_machine_ms: 

297 time spent executing the simulation on the machine actively using it 

298 :param resetting_ms: time spent resetting the simulation 

299 :param shutting_down_ms: time spent shutting down the simulation 

300 :param version: the version of the machine 

301 :param n_chips: number of chips that make up the machine 

302 :param n_active_chips: number of chips active in simulation 

303 :param n_boards: number of boards that make up the machine 

304 :param n_frames: number of frames that make up the machine 

305 :param n_cores: number of cores that are used by the simulation 

306 :param n_active_cores: number of cores actively used by the simulation 

307 :param load_chip_active_time: 

308 time that each core was active during loading 

309 :param extraction_chip_active_time: 

310 time that each core was active during extraction 

311 :param run_chip_active_time: 

312 time that each core was active during running 

313 :param load_router_packets: 

314 packets sent by the machine during loading 

315 :param extraction_router_packets: 

316 packets sent by the machine during extraction 

317 :param run_router_packets: 

318 packets sent by the machine during running 

319 """ 

320 

321 # Time and energy spent on the host machine, with the machine (at least 

322 # theoretically) running, doing general software tasks that we don't want 

323 # to put in other categories. 

324 other_time_s = ( 

325 waiting_ms + setup_ms + get_machine_ms + shutting_down_ms + 

326 run_other_ms + run_loop_ms + resetting_ms) / _MS_PER_SECOND 

327 other_energy_j = version.get_idle_energy( 

328 other_time_s, n_frames, n_boards, n_chips) 

329 

330 # Time and energy mapping to the machine 

331 mapping_time_s = mapping_ms / _MS_PER_SECOND 

332 mapping_energy_j = version.get_idle_energy( 

333 mapping_time_s, n_frames, n_boards, n_chips) 

334 

335 # Time and energy spent loading data onto the machine 

336 loading_time_s = ( 

337 loading_ms + data_loading_ms + expansion_ms) / _MS_PER_SECOND 

338 loading_energy_j = version.get_idle_energy( 

339 loading_ms / _MS_PER_SECOND, n_frames, n_boards, n_chips) 

340 loading_energy_j += version.get_active_energy( 

341 (data_loading_ms + expansion_ms) / _MS_PER_SECOND, n_frames, n_boards, 

342 n_chips, load_chip_active_time, load_router_packets) 

343 

344 # Time and energy spent extracting data from the machine 

345 saving_time_s = data_extraction_ms / _MS_PER_SECOND 

346 saving_energy_j = version.get_active_energy( 

347 saving_time_s, n_frames, n_boards, n_chips, 

348 extraction_chip_active_time, extraction_router_packets) 

349 

350 # Time and energy spent running the simulation on the machine 

351 exec_time_s = execute_on_machine_ms / _MS_PER_SECOND 

352 exec_energy_j = version.get_active_energy( 

353 exec_time_s, n_frames, n_boards, n_chips, 

354 run_chip_active_time, run_router_packets) 

355 exec_energy_cores_j = version.get_active_energy( 

356 exec_time_s, 0, 0, n_active_chips, run_chip_active_time, 

357 run_router_packets) 

358 exec_energy_boards_j = version.get_active_energy( 

359 exec_time_s, 0, n_boards, n_chips, run_chip_active_time, 

360 run_router_packets) 

361 

362 return PowerUsed( 

363 n_chips, n_active_chips, n_cores, n_active_cores, n_boards, n_frames, 

364 exec_time_s, mapping_time_s, loading_time_s, saving_time_s, 

365 other_time_s, exec_energy_j, exec_energy_cores_j, exec_energy_boards_j, 

366 mapping_energy_j, loading_energy_j, saving_energy_j, other_energy_j)
