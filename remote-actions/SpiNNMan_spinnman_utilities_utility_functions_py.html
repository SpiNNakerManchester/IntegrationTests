28def work_out_bmp_from_machine_details(hostname, number_of_boards): 

29 """ Work out the BMP connection IP address given the machine details.\ 

30 This is assumed to be the IP address of the machine, with 1 subtracted\ 

31 from the final part e.g. if the machine IP address is 192.168.0.5, the\ 

32 BMP IP address is assumed to be 192.168.0.4 

33 

34 :param str hostname: the SpiNNaker machine main hostname or IP address 

35 :param int number_of_boards: the number of boards in the machine 

36 :return: The BMP connection data 

37 :rtype: BMPConnectionData 

38 """ 

39 # take the IP address, split by dots, and subtract 1 off last bit 

40 ip_bits = socket.gethostbyname(hostname).split(".") 

41 ip_bits[-1] = str(int(ip_bits[-1]) - 1) 

42 bmp_ip_address = ".".join(ip_bits) 

43 

44 # add board scope for each split 

45 # if None, the end user didn't enter anything, so assume one board 

46 # starting at position 0 

47 if number_of_boards == 0 or number_of_boards is None: 

48 board_range = [0] 

49 else: 

50 board_range = range(number_of_boards) 

51 

52 # Assume a single board with no cabinet or frame specified 

53 return BMPConnectionData(cabinet=0, frame=0, ip_address=bmp_ip_address, 

54 boards=board_range, port_num=SCP_SCAMP_PORT)

57def get_vcpu_address(p): 

58 """ Get the address of the vcpu_t structure for the given core 

59 

60 :param int p: The core 

61 :rtype: int 

62 """ 

63 return CPU_INFO_OFFSET + (CPU_INFO_BYTES * p)

66def send_port_trigger_message(connection, board_address): 

67 """ Sends a port trigger message using a connection to (hopefully) open a\ 

68 port in a NAT and/or firewall to allow incoming packets to be received. 

69 

70 :param UDPConnection connection: 

71 The UDP connection down which the trigger message should be sent 

72 :param str board_address: 

73 The IP address of the SpiNNaker board to which the message should be 

74 sent 

75 """ 

76 

77 # Set up the message so that no reply is expected and it is sent to an 

78 # invalid port for SCAMP. The current version of SCAMP will reject 

79 # this message, but then fail to send a response since the 

80 # REPLY_NOT_EXPECTED flag is set (see scamp-3.c line 728 and 625-644) 

81 trigger_message = SDPMessage(SDPHeader( 

82 flags=SDPFlag.REPLY_NOT_EXPECTED, tag=0, destination_port=3, 

83 destination_cpu=0, destination_chip_x=0, destination_chip_y=0)) 

84 update_sdp_header_for_udp_send(trigger_message.sdp_header, 0, 0) 

85 connection.send_to( 

86 trigger_message.bytestring, (board_address, SCP_SCAMP_PORT))

89def reprogram_tag(connection, tag, strip=True): 

90 """ Reprogram an IP Tag to send responses to a given SCAMPConnection 

91 

92 :param SCAMPConnection connection: The connection to target the tag at 

93 :param int tag: The id of the tag to set 

94 :param bool strip: 

95 True if the tag should strip SDP headers from outgoing messages 

96 """ 

97 request = IPTagSet( 

98 connection.chip_x, connection.chip_y, [0, 0, 0, 0], 0, tag, 

99 strip=strip, use_sender=True) 

100 data = connection.get_scp_data(request) 

101 einfo = None 

102 for _ in range(3): 

103 try: 

104 connection.send(data) 

105 _, _, response, offset = \ 

106 connection.receive_scp_response() 

107 request.get_scp_response().read_bytestring(response, offset) 

108 return 

109 except SpinnmanTimeoutException: 

110 einfo = sys.exc_info() 

111 reraise(*einfo)
